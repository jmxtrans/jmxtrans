27  import com.googlecode.jmxtrans.JmxTransformer;
28  import com.googlecode.jmxtrans.cli.JmxTransConfiguration;
29  import com.googlecode.jmxtrans.guice.JmxTransModule;
30  import com.googlecode.jmxtrans.model.JmxProcess;
31  import com.googlecode.jmxtrans.model.Query;
32  import com.googlecode.jmxtrans.model.Server;
33  import com.googlecode.jmxtrans.model.output.GraphiteWriter;
34  import com.googlecode.jmxtrans.model.output.StdOutWriter;
35  
36  import java.util.Collections;
37  import java.util.Map;
38  
39  import static com.google.common.collect.Maps.newHashMap;
40  
41  /**
42   * This class produces the json that is in example.json.
43   * 
44   * @author jon
45   */
46  @SuppressWarnings({"squid:S106", "squid:S1118"}) // using StdOut if fine in an example
47  public class InterestingInfo {
48  
49  	private static final JsonPrinter printer = new JsonPrinter(System.out);
50  
51  	@SuppressWarnings("squid:S1313") // this is an example, no need to make IP addresses configurable
52  		public static void main(String[] args) throws Exception {
53  		Server.Builder serverBuilder = Server.builder()
54  				.setHost("w2")
55  				.setPort("1099")
56  				.setNumQueryThreads(2);
57  
58  		Map<String, Object> settings = newHashMap();
59  		settings.put(GraphiteWriter.HOST, "192.168.192.133");
60  		settings.put(GraphiteWriter.PORT, 2003);
61  
62  		GraphiteWriter gw = GraphiteWriter.builder()
63  				.setHost("192.168.192.133")
64  				.setPort(2003)
65  				.build();
66  
67  		StdOutWriter sw = new StdOutWriter(ImmutableList.<String>of(), false, false, Collections.<String, Object>emptyMap());
68  
69  		Query q = Query.builder()
70  				.setObj("java.lang:type=Memory")
71  				.addAttr("HeapMemoryUsage")
72  				.addAttr("NonHeapMemoryUsage")
73  				.addOutputWriterFactories(gw, sw)
74  				.build();
75  		serverBuilder.addQuery(q);
76  
77  		Query q2 = Query.builder()
78  				.setObj("java.lang:type=Threading")
79  				.addAttr("DaemonThreadCount")
80  				.addAttr("PeakThreadCount")
81  				.addAttr("ThreadCount")
82  				.addOutputWriterFactories(gw, sw)
83  				.build();
84  		serverBuilder.addQuery(q2);
85  
86  		Query q3 = Query.builder()
87  				.setObj("java.lang:name=ConcurrentMarkSweep,type=GarbageCollector")
88  				.addAttr("LastGcInfo")
89  				.addOutputWriterFactories(gw, sw)
90  				.build();
91  		serverBuilder.addQuery(q3);
92  
93  		Query q4 = Query.builder()
94  				.setObj("java.lang:name=ParNew,type=GarbageCollector")
95  				.addAttr("LastGcInfo")
96  				.addOutputWriterFactories(gw, sw)
97  				.build();
98  		serverBuilder.addQuery(q4);
99  
100 		JmxProcess process = new JmxProcess(serverBuilder.build());
101 		printer.prettyPrint(process);
102 		Injector injector = JmxTransModule.createInjector(new JmxTransConfiguration());
103 		JmxTransformer transformer = injector.getInstance(JmxTransformer.class);
104 
105 		transformer.executeStandalone(process);
106 	}
107 
108 }
