49  /**
50   * Walks a JMX tree and prints out all of the unique typenames and their
51   * attributes.
52   * 
53   * This is a good test of the core engine of JmxTrans to ensure that it covers
54   * all cases.
55   * 
56   * @author jon
57   */
58  public class TreeWalker3 {
59  
60  	private static final Logger log = LoggerFactory.getLogger(TreeWalker3.class);
61  
62  	/** */
63  	public static void main(String[] args) throws Exception {
64  		Server server = Server.builder().setHost("w2").setPort("1105").build();
65  
66  		JMXConnector conn = null;
67  		try {
68  			conn = server.getServerConnection();
69  			MBeanServerConnection mbeanServer = conn.getMBeanServerConnection();
70  
71  			TreeWalker3 tw = new TreeWalker3();
72  			tw.walkTree(mbeanServer, server);
73  		} catch (IOException e) {
74  			log.error("Problem processing queries for server: " + server.getHost() + ":" + server.getPort(), e);
75  		} finally {
76  			if (conn != null) {
77  				conn.close();
78  			}
79  		}
80  	}
81  
82  	public void walkTree(MBeanServerConnection connection, Server server) throws Exception {
83  
84  		// key here is null, null returns everything!
85  		Set<ObjectName> mbeans = connection.queryNames(null, null);
86  
87  		Map<String, String> output = newHashMap();
88  
89  		for (ObjectName name : mbeans) {
90  			MBeanInfo info = connection.getMBeanInfo(name);
91  			MBeanAttributeInfo[] attrs = info.getAttributes();
92  
93  			Query.Builder queryBuilder = Query.builder()
94  					.setObj(name.getCanonicalName());
95  			ResultCapture resultCapture = new ResultCapture();
96  			queryBuilder.addOutputWriterFactory(resultCapture);
97  
98  			for (MBeanAttributeInfo attrInfo : attrs) {
99  				queryBuilder.addAttr(attrInfo.getName());
100 			}
101 
102 			Query query = queryBuilder.build();
103 
104 			try {
105 				Iterable<Result> results = server.execute(query);
106 				query.runOutputWritersForQuery(server, results);
107 			} catch (AttributeNotFoundException anfe) {
108 				log.error("Error", anfe);
109 			}
110 
111 			for (Result result : resultCapture.results) {
112 				output.put(result.getTypeName(), query.getAttr().toString());
113 			}
114 		}
115 
116 		for (Entry<String, String> entry : output.entrySet()) {
117 			log.debug(entry.getKey());
118 			log.debug(entry.getValue());
119 			log.debug("-----------------------------------------");
120 		}
121 	}
122 
123 	private static final class ResultCapture extends OutputWriterAdapter implements OutputWriterFactory {
124 
125 		private List<Result> results;
126 
127 		@Override
128 		public void doWrite(Server server, Query query, Iterable<Result> results) throws Exception {
129 			this.results = copyOf(results);
130 		}
131 
132 		@Override
133 		public OutputWriter create() {
134 			return this;
135 		}
136 	}
137 }
