86  	/**
87  	 * Sets velocity up to load resources from a list of paths.
88  	 */
89  	protected VelocityEngine getVelocityEngine(List<String> paths) {
90  		VelocityEngine ve = new VelocityEngine();
91  		ve.setProperty(RuntimeConstants.RESOURCE_LOADER, "file");
92  		ve.setProperty("cp.resource.loader.class", "org.apache.velocity.runtime.resource.loader.FileResourceLoader");
93  		ve.setProperty("cp.resource.loader.cache", "true");
94  		ve.setProperty("cp.resource.loader.path", StringUtils.join(paths, ","));
95  		ve.setProperty("cp.resource.loader.modificationCheckInterval ", "10");
96  		ve.setProperty("input.encoding", "UTF-8");
97  		ve.setProperty("output.encoding", "UTF-8");
98  		ve.setProperty("runtime.log", "");
99  		return ve;
