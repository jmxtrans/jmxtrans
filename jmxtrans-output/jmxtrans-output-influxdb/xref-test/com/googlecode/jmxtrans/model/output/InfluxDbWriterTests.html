88  	@Test
89  	public void pointsAreWrittenToInfluxDb() throws Exception {
90  		InfluxDbWriter writer = getTestInfluxDbWriterWithDefaultSettings();
91  		writer.doWrite(dummyServer(), dummyQuery(), results);
92  
93  		verify(influxDB).write(messageCaptor.capture());
94  		BatchPoints batchPoints = messageCaptor.getValue();
95  
96  		assertThat(batchPoints.getDatabase()).isEqualTo(DATABASE_NAME);
97  
98  		// Point only exposes its state via a line protocol so we have to
99  		// make assertions against this.
100 		// Format is:
101 		// measurement,<comma separated key=val tags>" " <comma separated
102 		// key=val fields>
103 		Map<String, String> expectedTags = new TreeMap<String, String>();
104 		expectedTags.put(ResultAttributes.ATTRIBUTE_NAME.getName() , result.getAttributeName());
105 		expectedTags.put(ResultAttributes.CLASS_NAME.getName(), result.getClassName());
106 		expectedTags.put(ResultAttributes.OBJ_DOMAIN.getName
107 					(), result.getObjDomain());
108 		expectedTags.put(TAG_HOSTNAME, HOST);
109 		String lineProtocol = buildLineProtocol(result.getKeyAlias(), expectedTags);
110 
111 		List<Point> points = batchPoints.getPoints();
112 		assertThat(points).hasSize(1);
113 
114 		Point point = points.get(0);
115 		assertThat(point.lineProtocol()).startsWith(lineProtocol);
116 	}

177 	@Test
178 	public void onlyRequestedResultPropertiesAreAppliedAsTags() throws Exception {
179 		for (ResultAttribute expectedResultTag : ResultAttributes.values()) {
180 			ImmutableSet<ResultAttribute> expectedResultTags = ImmutableSet.of(expectedResultTag);
181 			InfluxDbWriter writer = getTestInfluxDbWriterWithResultTags(expectedResultTags);
182 			writer.doWrite(dummyServer(), dummyQuery(), results);
183 
184 			verify(influxDB, atLeastOnce()).write(messageCaptor.capture());
185 			BatchPoints batchPoints = messageCaptor.getValue();
186 			String lineProtocol = batchPoints.getPoints().get(0).lineProtocol();
187 
188 			assertThat(lineProtocol).contains(expectedResultTag.getName());
189 			Set<ResultAttribute> unexpectedResultTags = new HashSet<ResultAttribute>(ResultAttributes.values());
190 			unexpectedResultTags.remove(expectedResultTag);
191 			for (ResultAttribute unexpectedResultTag : unexpectedResultTags) {
192 				assertThat(lineProtocol).doesNotContain(unexpectedResultTag.getName());
193 			}
194 		}

222 	private String buildLineProtocol(String measurement, Map<String, String> expectedTags) {
223 		StringBuilder sb = new StringBuilder(measurement).append(",");
224 		int loops = 0;
225 		int tagCount = expectedTags.size();
226 		for (Map.Entry<String, String> entry : expectedTags.entrySet()) {
227 			loops++;
228 			sb.append(entry.getKey()).append("=").append(entry.getValue());
229 			if (loops < tagCount) {
230 				sb.append(",");
231 			}
232 		}
233 		return sb.toString();
234 	}
