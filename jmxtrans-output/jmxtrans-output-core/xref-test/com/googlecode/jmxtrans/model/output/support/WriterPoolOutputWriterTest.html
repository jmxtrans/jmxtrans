62  @RunWith(MockitoJUnitRunner.class)
63  public class WriterPoolOutputWriterTest {
64  
65  	private LifecycledPool<WriterPoolable> pool;
66  	@Spy private DummyAllocator allocator = new DummyAllocator();
67  	@Mock private WriterBasedOutputWriter target;
68  	private Writer writer = new StringWriter();
69  	@Captor private ArgumentCaptor<Writer> writerCaptor;
70  
71  	@Before
72  	public void setupPool() {
73  		Config<WriterPoolable> config = new Config<WriterPoolable>()
74  				.setAllocator(allocator);
75  		pool = new BlazePool<>(config);
76  	}
77  
78  	@Test
79  	public void writerIsPassedToTargetOutputWriter() throws Exception {
80  		WriterPoolOutputWriter<WriterBasedOutputWriter> outputWriter = new WriterPoolOutputWriter<>(target, pool, new Timeout(1, SECONDS));
81  
82  		outputWriter.doWrite(dummyServer(), dummyQuery(), dummyResults());
83  
84  		verify(target).write(writerCaptor.capture(), any(Server.class), any(Query.class), any(ImmutableList.class));
85  
86  		assertThat(writerCaptor.getValue()).isSameAs(writer);
87  	}
88  
89  	@Test(expected = IllegalStateException.class)
90  	public void claimTimeout() throws Exception {
91  		Timeout sleepTime = new Timeout(10, SECONDS);
92  		Timeout claimTime = new Timeout(1, SECONDS);
93  
94  		Config<WriterPoolable> config = new Config<WriterPoolable>()
95  			.setAllocator(new TimeoutAllocator(sleepTime));
96  		pool = new BlazePool<>(config);
97  
98  		WriterPoolOutputWriter<WriterBasedOutputWriter> outputWriter = new WriterPoolOutputWriter<>(target, pool, claimTime);
99  
100 		outputWriter.doWrite(dummyServer(), dummyQuery(), dummyResults());
101 	}
102 
103 	@Test(expected = IOException.class)
104 	@Category(IntegrationTest.class)
105 	public void poolableIsReleasedOnIOException() throws Exception {
106 		doThrow(IOException.class).when(target).write(any(Writer.class), any(Server.class), any(Query.class), any(ImmutableList.class));
107 
108 		WriterPoolOutputWriter<WriterBasedOutputWriter> outputWriter = new WriterPoolOutputWriter<>(target, pool, new Timeout(1, SECONDS));
109 		try {
110 			outputWriter.doWrite(dummyServer(), dummyQuery(), dummyResults());
111 		} finally {
112 			verify(allocator, timeout(500)).deallocate(any(WriterPoolable.class));
113 		}
114 	}
115 
116 	private class DummyAllocator implements Allocator<WriterPoolable> {
117 		@Override
118 		public WriterPoolable allocate(Slot slot) throws Exception {
119 			return new SocketPoolable(slot, null, writer, new NeverFlush());
120 		}
121 
122 		@Override
123 		public void deallocate(WriterPoolable poolable) throws Exception {
124 			poolable.release();
125 		}
126 	}
127 
128 	private class TimeoutAllocator implements Allocator<WriterPoolable> {
129 
130 		private Timeout timeout;
131 
132 		public TimeoutAllocator(Timeout timeout) {
133 			this.timeout = timeout;
134 		}
135 
136 		@Override
137 		public WriterPoolable allocate(Slot slot) throws Exception {
138 			try {
139 				Thread.sleep(timeout.getUnit().toMillis(timeout.getTimeout()));
140 			} catch (InterruptedException ex) {
141 				Thread.currentThread().interrupt();
142 			}
143 
144 			return new SocketPoolable(slot, null, writer, new NeverFlush());
145 		}
146 
147 		@Override
148 		public void deallocate(WriterPoolable poolable) throws Exception {
149 
150 		}
151 	}
152 }
