50  /**
51   * <a href="http://sensuapp.org/docs/0.12/events">Sensu Event Data</a>
52   * Format from <a href=https://github.com/SimpleFinance/chef-handler-sensu-event">chef-handler-sensu-event</a>
53   * Class structure from LibratoWriter
54   * <p/>
55   * Settings:
56   * <ul>
57   * <li>"{@code host}": Sensu client host. Optional, default value: {@value #DEFAULT_SENSU_HOST}</li>
58   * <li>"{@code handler}": Sensu handler. Optional, default value: {@value #DEFAULT_SENSU_HANDLER}</li>
59   * </ul>
60   *
61   * @author <a href="mailto:jhmartin@toger.us">Jason Martin</a>
62   */
63  @EqualsAndHashCode(exclude = "jsonFactory")
64  @ToString
65  public class SensuWriter extends BaseOutputWriter {
66  
67  	private static final Logger logger = LoggerFactory.getLogger(SensuWriter.class);
68  
69  	public static final String SETTING_HANDLER = "handler";
70  	public static final String DEFAULT_SENSU_HOST = "localhost";
71  	public static final String DEFAULT_SENSU_HANDLER = "graphite";
72  
73  	private final JsonFactory jsonFactory = new JsonFactory();
74  
75  	/** Sensu HTTP API URL */
76  	private final String host;
77  	private final String handler;
78  
79  	@JsonCreator
80  	public SensuWriter(
81  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
82  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
83  			@JsonProperty("debug") Boolean debugEnabled,
84  			@JsonProperty("host") String host,
85  			@JsonProperty("handler") String handler,
86  			@JsonProperty("settings") Map<String, Object> settings) {
87  		super(typeNames, booleanAsNumber, debugEnabled, settings);
88  		logger.warn("SensuWriter is deprecated. Please use SensuWriterFactory instead.");
89  		this.host = firstNonNull(host, (String) getSettings().get(HOST), DEFAULT_SENSU_HOST);
90  		this.handler = firstNonNull(handler, (String) getSettings().get(SETTING_HANDLER), DEFAULT_SENSU_HANDLER);
91  	}
92  
93  	@Override
94  	public void validateSetup(Server server, Query query) throws ValidationException {
95  		logger.info("Start Sensu writer connected to '{}' with handler {}", host, handler);
96  	}
97  
98  	@Override
99  	public void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
100 		logger.debug("Export to '{}', metrics {}", host, query);
101 		writeToSensu(server, query, results);
102 	}
103 
104 	private void serialize(Server server, Query query, List<Result> results, OutputStream outputStream) throws IOException {
105 		JsonGenerator g = jsonFactory.createGenerator(outputStream, JsonEncoding.UTF8);
106 		g.useDefaultPrettyPrinter();
107 		g.writeStartObject();
108 		g.writeStringField("name", "jmxtrans");
109 		g.writeStringField("type", "metric");
110 		g.writeStringField("handler", handler);
111 
112 		StringBuilder jsonoutput = new StringBuilder();
113 		List<String> typeNames = getTypeNames();
114 		for (Result result : results) {
115 			if (isNumeric(result.getValue())) {
116 				Object value = result.getValue();
117 				jsonoutput.append(KeyUtils.getKeyString(server, query, result, typeNames, null)).append(" ")
118 						.append(value).append(" ")
119 						.append(TimeUnit.SECONDS.convert(result.getEpoch(), TimeUnit.MILLISECONDS))
120 						.append(System.getProperty("line.separator"));
121 			}
122 		}
123 		g.writeStringField("output", jsonoutput.toString());
124 		g.writeEndObject();
125 		g.flush();
126 		g.close();
127 	}
128 
129 	private void writeToSensu(Server server, Query query, List<Result> results) {
130 		try (Socket socketConnection = new Socket(host, 3030)) {
131 			serialize(server, query, results, socketConnection.getOutputStream());
132 		} catch (Exception e) {
133 			logger.warn("Failure to send result to Sensu server '{}'", host, e);
134 		}
135 	}
136 
137 	public String getHost() {
138 		return host;
139 	}
140 
141 	public String getHandler() {
142 		return handler;
143 	}
