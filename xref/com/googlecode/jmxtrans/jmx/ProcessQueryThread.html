3   import com.googlecode.jmxtrans.model.Query;
4   import com.googlecode.jmxtrans.model.Server;
5   import org.slf4j.Logger;
6   import org.slf4j.LoggerFactory;
7   
8   import javax.management.MBeanServerConnection;
9   
10  /**
11   * Executes either a getAttribute or getAttributes query.
12   */
13  public class ProcessQueryThread implements Runnable {
14  
15  	private final Logger log = LoggerFactory.getLogger(getClass());
16  
17  	private final MBeanServerConnection mbeanServer;
18  	private final Server server;
19  	private final Query query;
20  
21  	public ProcessQueryThread(MBeanServerConnection mbeanServer, Server server, Query query) {
22  		this.mbeanServer = mbeanServer;
23  		this.server = server;
24  		this.query = query;
25  	}
26  
27  	public void run() {
28  		try {
29  			new JmxQueryProcessor().processQuery(this.mbeanServer, this.server, this.query);
30  		} catch (Exception e) {
31  			log.error("Error executing query: " + query, e);
32  			throw new RuntimeException(e);
33  		}
34  	}
35  }
