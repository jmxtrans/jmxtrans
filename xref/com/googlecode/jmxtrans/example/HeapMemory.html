23  package com.googlecode.jmxtrans.example;
24  
25  import com.google.inject.Injector;
26  import com.googlecode.jmxtrans.JmxTransformer;
27  import com.googlecode.jmxtrans.cli.JmxTransConfiguration;
28  import com.googlecode.jmxtrans.guice.JmxTransModule;
29  import com.googlecode.jmxtrans.model.JmxProcess;
30  import com.googlecode.jmxtrans.util.ProcessConfigUtils;
31  
32  import java.io.File;
33  
34  /**
35   * Shows how to process a file.
36   *
37   * @author jon
38   */
39  @SuppressWarnings({"squid:S106", "squid:S1118"}) // using StdOut if fine in an example
40  public class HeapMemory {
41  
42  	public static void main(String[] args) throws Exception {
43  		Injector injector = JmxTransModule.createInjector(new JmxTransConfiguration());
44  
45  		ProcessConfigUtils processConfigUtils = injector.getInstance(ProcessConfigUtils.class);
46  
47  		JmxProcess process = processConfigUtils.parseProcess(new File("heapmemory.json"));
48  		new JsonPrinter(System.out).print(process);
49  
50  		JmxTransformer transformer = injector.getInstance(JmxTransformer.class);
51  		transformer.executeStandalone(process);
52  	}
53  }
