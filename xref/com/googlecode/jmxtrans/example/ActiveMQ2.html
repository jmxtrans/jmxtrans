1   package com.googlecode.jmxtrans.example;
2   
3   import com.google.inject.Guice;
4   import com.google.inject.Injector;
5   import com.googlecode.jmxtrans.JmxTransformer;
6   import com.googlecode.jmxtrans.cli.JmxTransConfiguration;
7   import com.googlecode.jmxtrans.guice.JmxTransModule;
8   import com.googlecode.jmxtrans.model.JmxProcess;
9   import com.googlecode.jmxtrans.model.Query;
10  import com.googlecode.jmxtrans.model.Server;
11  import com.googlecode.jmxtrans.model.output.RRDToolWriter;
12  import com.googlecode.jmxtrans.util.JsonPrinter;
13  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
14  
15  import java.io.File;
16  import java.io.IOException;
17  
18  /**
19   * This example shows how to query an ActiveMQ server for some information.
20   * 
21   * The point of this example is to show that * works as part of the objectName.
22   * It also shows that you don't have to set an attribute to get for a query.
23   * jmxtrans will get all attributes on an object if you don't specify any.
24   * 
25   * @author jon
26   */
27  public class ActiveMQ2 {
28  
29  	private static final JsonPrinter jsonPrinter = new JsonPrinter(System.out);
30  
31  	@SuppressFBWarnings(
32  			value = "DMI_HARDCODED_ABSOLUTE_FILENAME",
33  			justification = "Path to RRD binary is hardcoded as this is example code")
34  	public static void main(String[] args) throws Exception {
35  		File outputFile = new File("target/w2-TEST.rrd");
36  		if (!outputFile.exists() && !outputFile.createNewFile()) {
37  			throw new IOException("Could not create output file");
38  		}
39  		RRDToolWriter gw = RRDToolWriter.builder()
40  				.setTemplateFile(new File("memorypool-rrd-template.xml"))
41  				.setOutputFile(outputFile)
42  				.setBinaryPath(new File("/opt/local/bin"))
43  				.setDebugEnabled(true)
44  				.setGenerate(true)
45  				.addTypeName("Destination")
46  				.build();
47  
48  		JmxProcess process = new JmxProcess(Server.builder()
49  				.setHost("w2")
50  				.setPort("1105")
51  				.setAlias("w2_activemq_1105")
52  				.addQuery(Query.builder()
53  						.setObj("org.apache.activemq:BrokerName=localhost,Type=Queue,Destination=*")
54  						.addAttr("QueueSize")
55  						.addAttr("MaxEnqueueTime")
56  						.addAttr("MinEnqueueTime")
57  						.addAttr("AverageEnqueueTime")
58  						.addAttr("InFlightCount")
59  						.addAttr("ConsumerCount")
60  						.addAttr("ProducerCount")
61  						.addAttr("DispatchCount")
62  						.addAttr("DequeueCount")
63  						.addAttr("EnqueueCount")
64  						.addOutputWriter(gw)
65  						.build())
66  				.addQuery(Query.builder()
67  						.setObj("org.apache.activemq:BrokerName=localhost,Type=Topic,Destination=*")
68  						.addAttr("QueueSize")
69  						.addAttr("MaxEnqueueTime")
70  						.addAttr("MinEnqueueTime")
71  						.addAttr("AverageEnqueueTime")
72  						.addAttr("InFlightCount")
73  						.addAttr("ConsumerCount")
74  						.addAttr("ProducerCount")
75  						.addAttr("DispatchCount")
76  						.addAttr("DequeueCount")
77  						.addAttr("EnqueueCount")
78  						.addOutputWriter(gw)
79  						.build()).build());
80  		jsonPrinter.prettyPrint(process);
81  
82  		Injector injector = Guice.createInjector(new JmxTransModule(new JmxTransConfiguration()));
83  		JmxTransformer transformer = injector.getInstance(JmxTransformer.class);
84  		transformer.executeStandalone(process);
85  	}
86  }
