16  import java.util.Collections;
17  import java.util.Map;

19  import static com.google.common.collect.Maps.newHashMap;

21  /**
22   * This class produces the json that is in example.json.
23   * 
24   * @author jon
25   */
26  public class InterestingInfo {
27  
28  	private static final JsonPrinter printer = new JsonPrinter(System.out);
29  
30  	public static void main(String[] args) throws Exception {
31  		Server.Builder serverBuilder = Server.builder()
32  				.setHost("w2")
33  				.setPort("1099")
34  				.setNumQueryThreads(2);
35  
36  		Map<String, Object> settings = newHashMap();
37  		settings.put(GraphiteWriter.HOST, "192.168.192.133");
38  		settings.put(GraphiteWriter.PORT, 2003);
39  
40  		GraphiteWriter gw = GraphiteWriter.builder()
41  				.setHost("192.168.192.133")
42  				.setPort(2003)
43  				.build();
44  
45  		StdOutWriter sw = new StdOutWriter(ImmutableList.<String>of(), false, false, Collections.<String, Object>emptyMap());
46  
47  		Query q = Query.builder()
48  				.setObj("java.lang:type=Memory")
49  				.addAttr("HeapMemoryUsage")
50  				.addAttr("NonHeapMemoryUsage")
51  				.addOutputWriters(gw, sw)
52  				.build();
53  		serverBuilder.addQuery(q);
54  
55  		Query q2 = Query.builder()
56  				.setObj("java.lang:type=Threading")
57  				.addAttr("DaemonThreadCount")
58  				.addAttr("PeakThreadCount")
59  				.addAttr("ThreadCount")
60  				.addOutputWriters(gw, sw)
61  				.build();
62  		serverBuilder.addQuery(q2);
63  
64  		Query q3 = Query.builder()
65  				.setObj("java.lang:name=ConcurrentMarkSweep,type=GarbageCollector")
66  				.addAttr("LastGcInfo")
67  				.addOutputWriters(gw, sw)
68  				.build();
69  		serverBuilder.addQuery(q3);
70  
71  		Query q4 = Query.builder()
72  				.setObj("java.lang:name=ParNew,type=GarbageCollector")
73  				.addAttr("LastGcInfo")
74  				.addOutputWriters(gw, sw)
75  				.build();
76  		serverBuilder.addQuery(q4);
77  
78  		JmxProcess process = new JmxProcess(serverBuilder.build());
79  		printer.prettyPrint(process);
80  		Injector injector = Guice.createInjector(new JmxTransModule(new JmxTransConfiguration()));
81  		JmxTransformer transformer = injector.getInstance(JmxTransformer.class);
82  
83  		transformer.executeStandalone(process);
84  	}
85  
86  }
