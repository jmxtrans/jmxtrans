1   package com.googlecode.jmxtrans.example;
2   
3   import com.google.inject.Guice;
4   import com.google.inject.Injector;
5   import com.googlecode.jmxtrans.JmxTransformer;
6   import com.googlecode.jmxtrans.cli.JmxTransConfiguration;
7   import com.googlecode.jmxtrans.guice.JmxTransModule;
8   import com.googlecode.jmxtrans.model.JmxProcess;
9   import com.googlecode.jmxtrans.model.Query;
10  import com.googlecode.jmxtrans.model.Server;
11  import com.googlecode.jmxtrans.model.output.GraphiteWriter;
12  import com.googlecode.jmxtrans.util.JsonPrinter;
13  
14  /**
15   * This example shows how to query hibernate for its statistics information.
16   * 
17   * @author jon
18   */
19  public class Hibernate {
20  
21  	private static final String GW_HOST = "192.168.192.133";
22  	private static final JsonPrinter printer = new JsonPrinter(System.out);
23  
24  	public static void main(String[] args) throws Exception {
25  
26  		JmxProcess process = new JmxProcess(Server.builder()
27  				.setHost("w2")
28  				.setPort("1099")
29  				.setAlias("w2_hibernate_1099")
30  				.addQuery(Query.builder()
31  					.setObj("org.hibernate.jmx:name=*,type=StatisticsService")
32  					.addAttr("EntityDeleteCount")
33  					.addAttr("EntityInsertCount")
34  					.addAttr("EntityLoadCount")
35  					.addAttr("EntityFetchCount")
36  					.addAttr("EntityUpdateCount")
37  					.addAttr("QueryExecutionCount")
38  					.addAttr("QueryCacheHitCount")
39  					.addAttr("QueryExecutionMaxTime")
40  					.addAttr("QueryCacheMissCount")
41  					.addAttr("QueryCachePutCount")
42  					.addAttr("FlushCount")
43  					.addAttr("ConnectCount")
44  					.addAttr("SecondLevelCacheHitCount")
45  					.addAttr("SecondLevelCacheMissCount")
46  					.addAttr("SecondLevelCachePutCount")
47  					.addAttr("SessionCloseCount")
48  					.addAttr("SessionOpenCount")
49  					.addAttr("CollectionLoadCount")
50  					.addAttr("CollectionFetchCount")
51  					.addAttr("CollectionUpdateCount")
52  					.addAttr("CollectionRemoveCount")
53  					.addAttr("CollectionRecreateCount")
54  					.addAttr("SuccessfulTransactionCount")
55  					.addAttr("TransactionCount")
56  					.addAttr("CloseStatementCount")
57  					.addAttr("PrepareStatementCount")
58  					.addAttr("OptimisticFailureCount")
59  					.addOutputWriter(GraphiteWriter.builder()
60  							.addTypeName("name")
61  							.setDebugEnabled(true)
62  							.setHost(GW_HOST)
63  							.setPort(2003)
64  							.build())
65  					.build())
66  				.build());
67  
68  		printer.prettyPrint(process);
69  
70  		Injector injector = Guice.createInjector(new JmxTransModule(new JmxTransConfiguration()));
71  		JmxTransformer transformer = injector.getInstance(JmxTransformer.class);
72  		transformer.executeStandalone(process);
73  
74  		// for (int i = 0; i < 160; i++) {
75  		// JmxUtils.processServer(server);
76  		// Thread.sleep(1000);
77  		// }
78  
79  	}
80  }
