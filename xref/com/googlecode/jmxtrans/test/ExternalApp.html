23  package com.googlecode.jmxtrans.test;
24  
25  import com.google.common.base.Function;
26  import com.google.common.base.Joiner;
27  import org.codehaus.mojo.animal_sniffer.IgnoreJRERequirement;
28  import org.junit.rules.ExternalResource;
29  
30  import javax.annotation.Nullable;
31  import java.io.File;
32  import java.io.InputStream;
33  import java.net.URISyntaxException;
34  import java.net.URL;
35  import java.net.URLClassLoader;
36  import java.util.ArrayList;
37  import java.util.List;
38  import java.util.Map;
39  import java.util.Properties;
40  
41  import static com.google.common.collect.FluentIterable.from;
42  import static java.util.Arrays.asList;
43  
44  public class ExternalApp extends ExternalResource {
45  
46  	private final String[] appArgs;
47  	private Process app;
48  	private final Class<?> appClass;
49  	private final Properties properties = new Properties();
50  	private File outputFile;
51  
52  	/**
53  	 * @param appClass Main class to run
54  	 * @param args Program arguments
55       */
56  	public ExternalApp(Class<?> appClass, String ... args) {
57  		this.appClass = appClass;
58  		this.appArgs = args;
59  	}
60  
61  	/**
62  	 * Enable JMX server in app
63       */
64  	public ExternalApp enableJmx(int jmxPort) {
65  		properties.put("com.sun.management.jmxremote.authenticate", "false");
66  		properties.put("com.sun.management.jmxremote.ssl", "false");
67  		properties.put("com.sun.management.jmxremote.port", Integer.toString(jmxPort));
68  		return this;
69  	}
70  
71  	public ExternalApp redirectOutputToFile(File outputFile) {
72  		this.outputFile = outputFile;
73  		return this;
74  	}
75  
76  	private String filePath(String filename) {
77  		URL fileUrl = getClass().getClassLoader().getResource(filename);
78  		try {
79  			return new File(fileUrl.toURI()).getAbsolutePath();
80  		} catch (URISyntaxException e) {
81  			return fileUrl.getPath();
82  		}
83  	}
84  
85  	public ExternalApp enableSsl(String keyStore, String keyStorePassword) {
86  		if (properties.get("com.sun.management.jmxremote.port") != null) {
87  			properties.put("com.sun.management.jmxremote.ssl", "true");
88  			properties.put("com.sun.management.jmxremote.registry.ssl", "true");
89  		}
90  		//properties.put("javax.net.debug", "ssl");
91  		String keyStorePath = filePath(keyStore);
92  		properties.put("javax.net.ssl.keyStore", keyStorePath);
93  		properties.put("javax.net.ssl.keyStorePassword", keyStorePassword);
94  		properties.put("javax.net.ssl.trustStore", keyStorePath);
95  		properties.put("javax.net.ssl.trustStorePassword", keyStorePassword);
96  		return this;
97  	}
98  
99  	@Override
100 	@IgnoreJRERequirement // ProcessBuilder.inheritIO() was introduced in Java 7. As this is only used in test, let's ignore it.
101 	protected void before() throws Throwable {
102 		List<String> command = new ArrayList<String>(asList("java", "-cp", getCurrentClasspath()));
103 		for(Map.Entry<Object, Object> property: properties.entrySet()) {
104 			command.add("-D"+property.getKey()+"="+property.getValue());
105 		}
106 		command.add(appClass.getCanonicalName());
107 		if (appArgs != null) {
108 			command.addAll(asList(appArgs));
109 		}
110 		ProcessBuilder processBuilder = new ProcessBuilder().command(command)
111 			.inheritIO();
112 		if (outputFile != null) {
113 			processBuilder.redirectOutput(outputFile);
114 		}
115 		app = processBuilder.start();
116 	}
117 
118 	public InputStream getInputStream() {
119 		return app.getInputStream();
120 	}
121 
122 	private String getCurrentClasspath() {
123 		ClassLoader cl = ClassLoader.getSystemClassLoader();
124 		URL[] urls = ((URLClassLoader)cl).getURLs();
125 
126 		return  Joiner.on(File.pathSeparator)
127 				.join(
128 						from(asList(urls))
129 								.transform(new Function<URL, String>() {
130 									@Nullable
131 									@Override
132 									public String apply(URL input) {
133 										try {
134 											return new File(input.toURI()).getPath();
135 										} catch(URISyntaxException e) {
136 											return input.getPath();
137 										}
138 									}
139 								}));
140 	}
141 
142 	@Override
143 	protected void after() {
144 		app.destroy();
145 	}
146 }
