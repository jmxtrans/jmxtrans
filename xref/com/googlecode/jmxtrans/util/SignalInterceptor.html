1   package com.googlecode.jmxtrans.util;
2   
3   import org.codehaus.mojo.animal_sniffer.IgnoreJRERequirement;
4   import sun.misc.Signal;
5   import sun.misc.SignalHandler;
6   
7   public abstract class SignalInterceptor {
8   
9   	protected SignalInterceptor() {
10  	}
11  
12  	/**
13  	 * <p>
14  	 * Register for the given signal. Note that the signal name should
15  	 * <b>not</b> begin with <b>SIG</b>. For example, if you are interested in
16  	 * <b>SIGTERM</b>, you should call <code>register("TERM")</code>.
17  	 * </p>
18  	 * <p>
19  	 * If the registration fails for any reason, a
20  	 * <code>SignalInterceptorException</code> will be thrown. This is usually
21  	 * caused by one of the following conditions:
22  	 * </p>
23  	 * <ul>
24  	 * <li>The <code>sun.misc.Signal*</code> classes are not available (e.g. you
25  	 * are not using Sun's JVM).</li>
26  	 * <li><code>signame</code> is not a valid trappable signal name on this OS
27  	 * (e.g. <b>KILL</b> can't be trapped, <b>HUP</b> does not exist on Windows)
28  	 * </li>
29  	 * <li>The JVM refuses to let you trap <code>signame</code> because it is
30  	 * already being used for some other important purpose (e.g. <b>QUIT</b>
31  	 * and/or <b>BREAK</b> cause the JVM to print diagnostic output).</li>
32  	 * </ul>
33  	 */
34  	protected void register(String signame) throws SignalInterceptorException {
35  		try {
36  			new SignalInterceptorHelper(signame, this);
37  		} catch (Throwable e) {
38  			throw new SignalInterceptorException(signame, e);
39  		}
40  	}
41  
42  	/**
43  	 * A wrapper around <code>register(String)</code> which never throws an
44  	 * exception. Instead, it returns <code>true</code> if registration
45  	 * succeeded, and <code>false</code> if it failed.
46  	 */
47  	protected boolean registerQuietly(String signame) {
48  		try {
49  			register(signame);
50  		} catch (Throwable e) {
51  			return false;
52  		}
53  		return true;
54  	}
55  
56  	/**
57  	 * Handle the given signal (which you had previously registered for). If
58  	 * this method return false, or throws an exception, subsequent handlers in
59  	 * the chain will <b>not</b> be called.
60  	 */
61  	protected abstract boolean handle(String signame);
62  
63  	/**
64  	 * <p>
65  	 * Private helper class for <code>SignalInterceptor</code>.
66  	 * </p>
67  	 * <p>
68  	 * This class exists separately from <code>SignalInterceptor</code> to
69  	 * permit graceful handling of LinkageErrors when the
70  	 * <code>sun.misc.Signal*</code> classes don't exist.
71  	 * </p>
72  	 */
73  	@IgnoreJRERequirement
74  	private static class SignalInterceptorHelper implements SignalHandler {
75  
76  		private final SignalHandler oldHandler;
77  
78  		private final SignalInterceptor interceptor;
79  
80  		@IgnoreJRERequirement
81  		SignalInterceptorHelper(String signame, SignalInterceptor interceptor) {
82  			this.interceptor = interceptor;
83  			Signal signal = new Signal(signame);
84  			oldHandler = Signal.handle(signal, this);
85  		}
86  
87  		@IgnoreJRERequirement
88  		public void handle(Signal sig) {
89  			if (interceptor.handle(sig.getName()) && (oldHandler != null)) {
90  				oldHandler.handle(sig);
91  			}
92  		}
93  	}
94  
95  	@SuppressWarnings("serial")
96  	private static class SignalInterceptorException extends Exception {
97  
98  		SignalInterceptorException(String signal, Throwable cause) {
99  			super("Unable to register for SIG" + signal, cause);
100 		}
101 
102 	}
103 }
