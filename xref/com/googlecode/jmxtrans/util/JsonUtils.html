3   import com.fasterxml.jackson.databind.ObjectMapper;
4   import com.fasterxml.jackson.datatype.guava.GuavaModule;
5   
6   import java.io.File;
7   import java.io.IOException;
8   
9   import com.googlecode.jmxtrans.model.JmxProcess;
10  
11  public final class JsonUtils {
12  
13  	private JsonUtils() {}
14  
15  	/**
16  	 * Uses jackson to load json configuration from a File into a full object
17  	 * tree representation of that json.
18  	 */
19  	public static JmxProcess getJmxProcess(File file) throws IOException {
20  		ObjectMapper mapper = new ObjectMapper();
21  		mapper.registerModule(new GuavaModule());
22  		JmxProcess jmx = mapper.readValue(file, JmxProcess.class);
23  		jmx.setName(file.getName());
24  		return jmx;
25  	}
26  }
