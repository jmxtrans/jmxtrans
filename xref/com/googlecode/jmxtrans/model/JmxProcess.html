1   package com.googlecode.jmxtrans.model;

3   import com.fasterxml.jackson.annotation.JsonIgnore;
4   import com.fasterxml.jackson.databind.annotation.JsonSerialize;
5   import com.google.common.collect.ImmutableList;
6   
7   import java.util.List;
8   
9   import static com.fasterxml.jackson.databind.annotation.JsonSerialize.Inclusion.NON_NULL;
10  
11  /**
12   * This is the container for a list of Servers.
13   *
14   * @author jon
15   */
16  @JsonSerialize(include = NON_NULL)
17  public class JmxProcess {
18  
19  	private String name;
20  	private ImmutableList<Server> servers;
21  	private Integer numMultiThreadedServers;
22  
23  	public JmxProcess() {
24  		this(ImmutableList.<Server>of());
25  	}
26  
27  	public JmxProcess(Server server) {
28  		this.servers = ImmutableList.of(server);
29  	}
30  
31  	public JmxProcess(ImmutableList<Server> servers) {
32  		this.setServers(servers);
33  	}
34  
35  	public void setServers(List<Server> servers) {
36  		this.servers = ImmutableList.copyOf(servers);
37  	}
38  
39  	public ImmutableList<Server> getServers() {
40  		return servers;
41  	}
42  
43  	@JsonIgnore
44  	public boolean isServersMultiThreaded() {
45  		return this.numMultiThreadedServers != null && this.numMultiThreadedServers > 0;
46  	}
47  
48  	/**
49  	 * Set this if you want each JmxProcess to run in its own thread up to the
50  	 * count you set. So, if you set this to 3 and you have 6 JmxProcesses,
51  	 * it'll start up a max of 3 threads and then block until a thread becomes
52  	 * available. It'll then start another thread up to fill in the blanks.
53  	 */
54  	public void setNumMultiThreadedServers(Integer numMultiThreadedServers) {
55  		this.numMultiThreadedServers = numMultiThreadedServers;
56  	}
57  
58  	public Integer getNumMultiThreadedServers() {
59  		return numMultiThreadedServers;
60  	}
61  
62  	public void setName(String name) {
63  		this.name = name;
64  	}
65  
66  	public String getName() {
67  		return name;
68  	}
69  }
