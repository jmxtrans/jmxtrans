102 	/**
103 	 * Generates the proper username/password environment for JMX connections.
104 	 */
105 	@JsonIgnore
106 	public ImmutableMap<String, ?> getEnvironment() {
107 		if (getProtocolProviderPackages() != null && getProtocolProviderPackages().contains("weblogic")) {
108 			ImmutableMap.Builder<String, String> environment = ImmutableMap.builder();
109 			if ((username != null) && (password != null)) {
110 				environment.put(PROTOCOL_PROVIDER_PACKAGES, getProtocolProviderPackages());
111 				environment.put(SECURITY_PRINCIPAL, username);
112 				environment.put(SECURITY_CREDENTIALS, password);
113 			}
114 			return environment.build();
115 		}
116 
117 		ImmutableMap.Builder<String, String[]> environment = ImmutableMap.builder();
118 		if ((username != null) && (password != null)) {
119 			String[] credentials = new String[] {
120 					username,
121 					password
122 			};
123 			environment.put(JMXConnector.CREDENTIALS, credentials);
124 		}
125 
126 		return environment.build();
127 	}

153 	public String getHost() {
154 		if (host == null && url == null) {
155 			// TODO: shouldn't we just return a null in this case ?
156 			throw new IllegalStateException("host is null and url is null. Cannot construct host dynamically.");
157 		}
158 
159 		if (host != null) {
160 			return host;
161 		}
162 
163 		// removed the caching of the extracted host as it is a very minor
164 		// optimization we should probably pre compute it in the builder and
165 		// throw exception at construction if both url and host are set
166 		// we might also be able to use java.net.URI to parse the URL, but I'm
167 		// not familiar enough with JMX URLs to think of the test cases ...
168 		return url.substring(url.lastIndexOf("//") + 2, url.lastIndexOf(':'));
169 	}
170 
171 	public String getPort() {
172 		if (port == null && url == null) {
173 			throw new IllegalStateException("port is null and url is null.  Cannot construct port dynamically.");
174 		}
175 		if (this.port != null) {
176 			return port;
177 		}
178 
179 		return extractPortFromUrl(url);
180 	}

182 	private static String extractPortFromUrl(String url) {
183 		String computedPort = url.substring(url.lastIndexOf(':') + 1);
184 		if (computedPort.contains("/")) {
185 			computedPort = computedPort.substring(0, computedPort.indexOf('/'));
186 		}
187 		return computedPort;
188 	}

211 	/**
212 	 * The jmx url to connect to. If null, it builds this from host/port with a
213 	 * standard configuration. Other JVM's may want to set this value.
214 	 */
215 	public String getUrl() {
216 		if (this.url == null) {
217 			if ((this.host == null) || (this.port == null)) {
218 				throw new RuntimeException("url is null and host or port is null. cannot construct url dynamically.");
219 			}
220 			return FRONT + this.host + ":" + this.port + BACK;
221 		}
222 		return this.url;
223 	}

431 	/**
432 	 * Merges two lists of servers (and their queries). Based on the equality of
433 	 * both sets of objects. Public for testing purposes.
434 	 * @param secondList
435 	 * @param firstList
436 	 */
437 	// FIXME: the params for this method should be Set<Server> as there are multiple assumptions that they are unique
438 	@CheckReturnValue
439 	public static ImmutableList<Server> mergeServerLists(List<Server> firstList, List<Server> secondList) {
440 		ImmutableList.Builder<Server> results = ImmutableList.builder();
441 		List<Server> toProcess = new ArrayList<Server>(secondList);
442 		for (Server firstServer : firstList) {
443 			if (toProcess.contains(firstServer)) {
444 				Server found = toProcess.get(secondList.indexOf(firstServer));
445 				results.add(merge(firstServer, found));
446 				// remove server as it is already merged
447 				toProcess.remove(found);
448 			} else {
449 				results.add(firstServer);
450 			}
451 		}
452 		// add servers from the second list that are not in the first one
453 		results.addAll(toProcess);
454 		return results.build();
455 	}
456 
457 	private static Server merge(Server firstServer, Server secondServer) {
458 		return builder(firstServer)
459 				.addQueries(secondServer.getQueries())
460 				.build();
461 	}
