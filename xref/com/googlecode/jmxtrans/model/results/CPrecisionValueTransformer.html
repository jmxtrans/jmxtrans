1   package com.googlecode.jmxtrans.model.results;
2   
3   import javax.annotation.Nullable;
4   import java.math.BigDecimal;
5   
6   public class CPrecisionValueTransformer implements ValueTransformer {
7   
8   	private static final BigDecimal C_PRECISION = new BigDecimal("1E-308");
9   
10  	@Nullable
11  	@Override
12  	public Object apply(Object input) {
13  		if (input == null) return null;
14  		if (!Number.class.isAssignableFrom(input.getClass())) return input;
15  
16  		BigDecimal inputNumber = new BigDecimal(input.toString());
17  
18  		if (inputNumber.abs().compareTo(C_PRECISION) < 0) return 0;
19  
20  		return input;
21  	}
22  
23  }
