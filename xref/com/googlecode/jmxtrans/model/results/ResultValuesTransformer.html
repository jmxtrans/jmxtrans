23  package com.googlecode.jmxtrans.model.results;
24  
25  import com.google.common.base.Function;
26  import com.googlecode.jmxtrans.model.Result;
27  
28  import javax.annotation.Nullable;
29  
30  public class ResultValuesTransformer implements Function<Result, Result> {
31  
32  	private final ValueTransformer valueTransformer;
33  
34  	public ResultValuesTransformer(ValueTransformer valueTransformer) {
35  		this.valueTransformer = valueTransformer;
36  	}
37  
38  	@Nullable
39  	@Override
40  	public Result apply(@Nullable Result input) {
41  		if (input == null) {
42  			return null;
43  		}
44  		return new Result(
45  				input.getEpoch(),
46  				input.getAttributeName(),
47  				input.getClassName(),
48  				input.getObjDomain(),
49  				input.getKeyAlias(),
50  				input.getTypeName(),
51  				input.getValuePath(),
52  				valueTransformer.apply(input.getValue())
53  		);
54  	}
55  
56  }
