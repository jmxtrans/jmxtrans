3   import javax.annotation.Nullable;
4   
5   public class BooleanAsNumberValueTransformer implements ValueTransformer {
6   
7   	private final Number valueForTrue;
8   	private final Number valueForFalse;
9   
10  	public BooleanAsNumberValueTransformer(Number valueForTrue, Number valueForFalse) {
11  		this.valueForTrue = valueForTrue;
12  		this.valueForFalse = valueForFalse;
13  	}
14  
15  	@Nullable
16  	@Override
17  	public Object apply(@Nullable Object value) {
18  		if (value == null) {
19  			return null;
20  		}
21  
22  		if (value instanceof Boolean) {
23  			if ((Boolean) value) {
24  				return valueForTrue;
25  			} else {
26  				return valueForFalse;
27  			}
28  		}
29  		return value;
30  	}
31  }
