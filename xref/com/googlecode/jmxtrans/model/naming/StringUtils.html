3   import java.util.regex.Pattern;
4   
5   public final class StringUtils {
6   	private static final Pattern DOT_SLASH_UNDERSCORE_PAT = Pattern.compile("[./]");
7   	private static final Pattern SLASH_UNDERSCORE_PAT = Pattern.compile("/", Pattern.LITERAL);
8   	private static final Pattern SPACE_PAT = Pattern.compile("[ \"']+");
9   
10  	private StringUtils() {}
11  
12  	/**
13  	 * Replaces all . and / with _ and removes all spaces and double/single quotes.
14  	 *
15  	 * @param name the name
16  	 * @return the string
17  	 */
18  	public static String cleanupStr(String name) {
19  		return cleanupStr(name, false);
20  	}
21  
22  	/**
23  	 * Replaces all . and / with _ and removes all spaces and double/single quotes.
24  	 *
25  	 * @param name            the name
26  	 * @param allowDottedKeys whether we remove the dots or not.
27  	 * @return
28  	 */
29  	public static String cleanupStr(String name, boolean allowDottedKeys) {
30  		if (name == null) {
31  			return null;
32  		}
33  		Pattern pattern;
34  		if (!allowDottedKeys) {
35  			pattern = DOT_SLASH_UNDERSCORE_PAT;
36  		} else {
37  			pattern = SLASH_UNDERSCORE_PAT;
38  		}
39  		String clean = pattern.matcher(name).replaceAll("_");
40  		clean = SPACE_PAT.matcher(clean).replaceAll("");
41  		return clean;
42  	}
43  }
