25  /**
26   * This takes a JRobin template.xml file and then creates the database if it
27   * doesn't already exist.
28   * 
29   * It will then write the contents of the Query (the Results) to the database.
30   * 
31   * This uses the JRobin rrd format and is incompatible with the C version of
32   * rrd.
33   * 
34   * @author jon
35   */
36  public class RRDWriter extends BaseOutputWriter {
37  
38  	private final File outputFile;
39  	private final File templateFile;
40  
41  	@JsonCreator
42  	public RRDWriter(
43  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
44  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
45  			@JsonProperty("debug") Boolean debugEnabled,
46  			@JsonProperty("outputFile") String outputFile,
47  			@JsonProperty("templateFile") String templateFile,
48  			@JsonProperty("settings") Map<String, Object> settings) {
49  		super(typeNames, booleanAsNumber, debugEnabled, settings);
50  		this.outputFile = new File(MoreObjects.firstNonNull(outputFile, (String) getSettings().get(OUTPUT_FILE)));
51  		this.templateFile = new File(MoreObjects.firstNonNull(templateFile, (String) getSettings().get(TEMPLATE_FILE)));
52  		checkState(this.outputFile.exists(), "Output file must exist");
53  		checkState(this.templateFile.exists(), "Template file must exist");
54  	}
55  
56  	public void validateSetup(Server server, Query query) throws ValidationException {
57  	}
58  
59  	public void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
60  		RrdDb db = null;
61  		try {
62  			db = createOrOpenDatabase();
63  			Sample sample = db.createSample();
64  			List<String> dsNames = Arrays.asList(db.getDsNames());
65  
66  			// go over all the results and look for datasource names that map to
67  			// keys from the result values
68  			for (Result res : results) {
69  				Map<String, Object> values = res.getValues();
70  				if (values != null) {
71  					for (Entry<String, Object> entry : values.entrySet()) {
72  						if (dsNames.contains(entry.getKey()) && NumberUtils.isNumeric(entry.getValue())) {
73  							sample.setValue(entry.getKey(), Double.valueOf(entry.getValue().toString()));
74  						}
75  					}
76  				}
77  			}
78  			sample.update();
79  		} finally {
80  			if (db != null) {
81  				db.close();
82  			}
83  		}
84  	}
85  
86  	/**
87  	 * If the database file doesn't exist, it'll get created, otherwise, it'll
88  	 * be returned in r/w mode.
89  	 */
90  	protected RrdDb createOrOpenDatabase() throws Exception {
91  		RrdDb result;
92  		if (!this.outputFile.exists()) {
93  			FileUtils.forceMkdir(this.outputFile.getParentFile());
94  			RrdDefTemplate t = new RrdDefTemplate(this.templateFile);
95  			t.setVariable("database", this.outputFile.getCanonicalPath());
96  			RrdDef def = t.getRrdDef();
97  			result = new RrdDb(def);
98  		} else {
99  			result = new RrdDb(this.outputFile.getCanonicalPath());
100 		}
101 		return result;
102 	}
103 
104 	public String getTemplateFile() {
105 		return templateFile.getPath();
106 	}
107 
108 	public String getOutputFile() {
109 		return outputFile.getPath();
110 	}
111 }
