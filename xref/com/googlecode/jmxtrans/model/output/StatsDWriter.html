33  /**
34   * This output writer sends data to a host/port combination in the StatsD
35   * format.
36   *
37   * @author neilh
38   */
39  public class StatsDWriter extends BaseOutputWriter {

42  	public static final String ROOT_PREFIX = "rootPrefix";

56  	/**
57  	 * Uses JmxUtils.getDefaultPoolMap()
58  	 *
59  	 * @throws IOException
60  	 */
61  	@JsonCreator
62  	public StatsDWriter(
63  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
64  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
65  			@JsonProperty("debug") Boolean debugEnabled,
66  			@JsonProperty("host") String host,
67  			@JsonProperty("port") Integer port,
68  			@JsonProperty("bucketType") String bucketType,
69  			@JsonProperty("rootPrefix") String rootPrefix,
70  			@JsonProperty("settings") Map<String, Object> settings) throws IOException {
71  		super(typeNames, booleanAsNumber, debugEnabled, settings);
72  		channel = DatagramChannel.open();
73  		sendBuffer = ByteBuffer.allocate((short) 1500);
74  
75  		// bucketType defaults to c == counter
76  		this.bucketType = firstNonNull(bucketType, (String) getSettings().get(BUCKET_TYPE), "c");
77  		this.rootPrefix = firstNonNull(rootPrefix, (String) getSettings().get(ROOT_PREFIX), "servers");
78  
79  		if (host == null) {
80  			host = (String) getSettings().get(HOST);
81  		}
82  
83  		if (port == null) {
84  			port = Settings.getIntegerSetting(getSettings(), PORT, null);
85  		}
86  		checkNotNull(host, "Host cannot be null");
87  		checkNotNull(port, "Port cannot be null");
88  		this.address = new InetSocketAddress(host, port);
89  	}

95  	@Override
96  	public void start() throws LifecycleException {
97  		try {
98  			pool = new GenericKeyedObjectPool<SocketAddress, DatagramSocket>(new DatagramSocketFactory());
99  			pool.setTestOnBorrow(true);
100 			pool.setMaxActive(-1);
101 			pool.setMaxIdle(-1);
102 			pool.setTimeBetweenEvictionRunsMillis(1000 * 60 * 5);
103 			pool.setMinEvictableIdleTimeMillis(1000 * 60 * 5);
104 
105 			this.mbean = new ManagedGenericKeyedObjectPool((GenericKeyedObjectPool) pool, "StatsdConnectionPool");
106 			ManagementFactory.getPlatformMBeanServer()
107 					.registerMBean(this.mbean, this.mbean.getObjectName());
108 		} catch (Exception e) {
109 			throw new LifecycleException(e);
110 		}
111 	}
112 
113 	@Override
114 	public void stop() throws LifecycleException {
115 		try {
116 			if (this.mbean != null) {
117 				MBeanServer mbs = ManagementFactory.getPlatformMBeanServer();
118 				mbs.unregisterMBean(this.mbean.getObjectName());
119 				this.mbean = null;
120 			}
121 			if (this.pool != null) {
122 				pool.close();
123 				this.pool = null;
124 			}
125 		} catch (Exception e) {
126 			throw new LifecycleException(e);
127 		}
128 	}

130 	public void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
131 
132 		List<String> typeNames = this.getTypeNames();
133 
134 		for (Result result : results) {
135 			if (isDebugEnabled()) {
136 				log.debug(result.toString());
137 			}
138 
139 			Map<String, Object> resultValues = result.getValues();
140 			if (resultValues != null) {
141 				for (Entry<String, Object> values : resultValues.entrySet()) {
142 					if (NumberUtils.isNumeric(values.getValue())) {
143 
144 						String line = KeyUtils.getKeyString(server, query, result, values, typeNames, rootPrefix)
145 								+ ":" + values.getValue().toString() + "|" + bucketType + "\n";
146 
147 						if (isDebugEnabled()) {
148 							log.debug("StatsD Message: " + line.trim());
149 						}
150 
151 						doSend(line.trim());
152 					}
153 				}
154 			}
155 		}
156 	}
157 
158 	private synchronized boolean doSend(String stat) {
159 		try {
160 			final byte[] data = stat.getBytes("utf-8");
161 
162 			// If we're going to go past the threshold of the buffer then flush.
163 			// the +1 is for the potential '\n' in multi_metrics below
164 			if (sendBuffer.remaining() < (data.length + 1)) {
165 				flush();
166 			}
167 
168 			if (sendBuffer.position() > 0) { // multiple metrics are separated
169 				// by '\n'
170 				sendBuffer.put((byte) '\n');
171 			}
172 
173 			sendBuffer.put(data); // append the data
174 
175 			flush();
176 			return true;
177 
178 		} catch (IOException e) {
179 			e.printStackTrace();
180 			return false;
181 		}
182 	}
183 
184 	public synchronized boolean flush() {
185 		try {
186 			final int sizeOfBuffer = sendBuffer.position();
187 
188 			if (sizeOfBuffer <= 0) {
189 				return false;
190 			} // empty buffer
191 
192 			// send and reset the buffer
193 			sendBuffer.flip();
194 			final int nbSentBytes = channel.send(sendBuffer, this.address);
195 			sendBuffer.limit(sendBuffer.capacity());
196 			sendBuffer.rewind();
197 
198 			return sizeOfBuffer == nbSentBytes;
199 
200 		} catch (IOException e) {
201 			e.printStackTrace();
202 			return false;
203 		}
204 	}
