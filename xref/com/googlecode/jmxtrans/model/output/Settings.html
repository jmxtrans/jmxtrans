3   import java.util.Map;
4   
5   // FIXME settings parsing is very tolerant. It will try its best to return
6   // default values, silently ignoring parsing errors. Except in the case of
7   // primitive int settings, where is raises an IllegalArgumentException. We
8   // should resolve this incoherence, probably by always raising an exception
9   // when the setting cannot be parsed.
10  // FIXME resolution of settings (via PropertyResolver.resolveMap()) is not
11  // consistent. Settings added by other means than setSettings() are never
12  // resolved.
13  public final class Settings {
14  	private Settings() {}
15  
16  	/**
17  	 * Returns a boolean value for the key, defaults to false if not specified.
18  	 * This is equivalent to calling {@link #getBooleanSetting(java.util.Map, String, Boolean)}.
19  	 *
20  	 *
21  	 * @param settings
22  	 * @param key the key value to get the boolean setting for
23  	 * @return the value set for the key, defaults to false
24  	 */
25  	public static Boolean getBooleanSetting(Map<String, Object> settings, String key) {
26  		return getBooleanSetting(settings, key, false);
27  	}
28  
29  	/**
30  	 * Gets a Boolean value for the key, returning the default value given if
31  	 * not specified or not a valid boolean value.
32  	 *
33  	 *
34  	 * @param settings
35  	 * @param key        the key to get the boolean setting for
36  	 * @param defaultVal the default value to return if the setting was not specified
37  	 *                   or was not coercible to a Boolean value
38  	 * @return the Boolean value for the setting
39  	 */
40  	public static Boolean getBooleanSetting(Map<String, Object> settings, String key, Boolean defaultVal) {
41  		final Object value = settings.get(key);
42  
43  		if (value == null) {
44  			return defaultVal;
45  		}
46  		if (value instanceof Boolean) {
47  			return (Boolean) value;
48  		}
49  		if (value instanceof String) {
50  			return Boolean.valueOf((String) value);
51  		}
52  		return defaultVal;
53  	}
54  
55  	/**
56  	 * Gets an Integer value for the key, returning the default value given if
57  	 * not specified or not a valid numeric value.
58  	 *
59  	 *
60  	 * @param settings
61  	 * @param key        the key to get the Integer setting for
62  	 * @param defaultVal the default value to return if the setting was not specified
63  	 *                   or was not coercible to an Integer value
64  	 * @return the Integer value for the setting
65  	 */
66  	public static Integer getIntegerSetting(Map<String, Object> settings, String key, Integer defaultVal) {
67  		final Object value = settings.get(key);
68  		if (value == null) {
69  			return defaultVal;
70  		}
71  
72  		if (value instanceof Number) {
73  			return ((Number) value).intValue();
74  		}
75  		if (value instanceof String) {
76  			try {
77  				return Integer.parseInt((String) value);
78  			} catch (NumberFormatException e) {
79  				return defaultVal;
80  			}
81  		}
82  		return defaultVal;
83  	}
84  
85  	/**
86  	 * Gets a String value for the setting, returning the default value if not
87  	 * specified.
88  	 *
89  	 *
90  	 * @param settings
91  	 * @param key        the key to get the String setting for
92  	 * @param defaultVal the default value to return if the setting was not specified
93  	 * @return the String value for the setting
94  	 */
95  	public static String getStringSetting(Map<String, Object> settings, String key, String defaultVal) {
96  		final Object value = settings.get(key);
97  		return value != null ? value.toString() : defaultVal;
98  	}
99  
100 	/**
101 	 * Gets an int value for the setting, returning the default value if not
102 	 * specified.
103 	 *
104 	 *
105 	 * @param settings
106 	 * @param key        the key to get the int value for
107 	 * @param defaultVal default value if the setting was not specified
108 	 * @return the int value for the setting
109 	 * @throws IllegalArgumentException if setting does not contain an int
110 	 */
111 	protected static int getIntSetting(Map<String, Object> settings, String key, int defaultVal) throws IllegalArgumentException {
112 		if (settings.containsKey(key)) {
113 			final Object objectValue = settings.get(key);
114 			if (objectValue == null) {
115 				throw new IllegalArgumentException("Setting '" + key + " null");
116 			}
117 			final String value = objectValue.toString();
118 			try {
119 				return Integer.parseInt(value);
120 			} catch (Exception e) {
121 				throw new IllegalArgumentException("Setting '" + key + "=" + value + "' is not an integer");
122 			}
123 		} else {
124 			return defaultVal;
125 		}
126 	}
127 }
