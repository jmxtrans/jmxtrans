27  /**
28   * Writes out data in the same format as the GraphiteWriter, except to a file
29   * and tab delimited. Takes advantage of Log4J RollingFileAppender to
30   * automatically handle rolling the files after they reach a certain size.
31   * <p/>
32   * The default max size of the log files are 10MB (maxLogFileSize) The default
33   * number of rolled files to keep is 200 (maxLogBackupFiles)
34   *
35   * @author jon
36   */
37  public class KeyOutWriter extends BaseOutputWriter {
38  
39  	protected static final Log4jLoggerFactory log4jLoggerFactory = new Log4jLoggerFactory();
40  	protected static final String SETTING_MAX_LOG_FILE_SIZE = "maxLogFileSize";
41  	protected static final String SETTING_MAX_BACK_FILES = "maxLogBackupFiles";
42  	protected static final String SETTING_DELIMITER = "delimiter";
43  	protected static final String LOG_PATTERN = "%m%n";
44  	protected static final int LOG_IO_BUFFER_SIZE_BYTES = 1024;
45  	protected static final Map<String, Logger> loggers = new ConcurrentHashMap<String, Logger>();
46  
47  	protected static final int MAX_LOG_BACKUP_FILES = 200;
48  	protected static final String MAX_LOG_FILE_SIZE = "10MB";
49  	protected static final String DEFAULT_DELIMITER = "\t";
50  	protected Logger logger;
51  
52  
53  	private final String outputFile;
54  	private final String maxLogFileSize;
55  	private final int maxLogBackupFiles;
56  	private final String delimiter;
57  
58  	@JsonCreator
59  	public KeyOutWriter(
60  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
61  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
62  			@JsonProperty("debug") Boolean debugEnabled,
63  			@JsonProperty("outputFile") String outputFile,
64  			@JsonProperty("maxLogFileSize") String maxLogFileSize,
65  			@JsonProperty("maxLogBackupFiles") int maxLogBackupFiles,
66  			@JsonProperty("delimiter") String delimiter,
67  			@JsonProperty("settings") Map<String, Object> settings) {
68  		super(typeNames, booleanAsNumber, debugEnabled, settings);
69  		this.outputFile = MoreObjects.firstNonNull(
70  				outputFile,
71  				(String) getSettings().get("outputFile"));
72  		this.maxLogFileSize = firstNonNull(
73  				maxLogFileSize,
74  				(String) getSettings().get(SETTING_MAX_LOG_FILE_SIZE),
75  				MAX_LOG_FILE_SIZE);
76  		this.maxLogBackupFiles = firstNonNull(
77  				maxLogBackupFiles,
78  				(Integer) getSettings().get(SETTING_MAX_BACK_FILES),
79  				MAX_LOG_BACKUP_FILES);
80  		this.delimiter = firstNonNull(
81  				delimiter,
82  				(String) getSettings().get(SETTING_DELIMITER),
83  				DEFAULT_DELIMITER
84  		);
85  	}
86  
87  	/**
88  	 * Creates the logging
89  	 */
90  	@Override
91  	public void validateSetup(Server server, Query query) throws ValidationException {
92  		// Check if we've already created a logger for this file. If so, use it.
93  		if (loggers.containsKey(outputFile)) {
94  			logger = loggers.get(outputFile);
95  			return;
96  		}
97  		// need to create a logger
98  		try {
99  			logger = initLogger(outputFile);
100 			loggers.put(outputFile, logger);
101 		} catch (IOException e) {
102 			throw new ValidationException("Failed to setup log4j", query);
103 		}
104 	}
105 
106 	/**
107 	 * The meat of the output. Very similar to GraphiteWriter.
108 	 */
109 	@Override
110 	public void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
111 		List<String> typeNames = getTypeNames();
112 
113 		for (Result result : results) {
114 			Map<String, Object> resultValues = result.getValues();
115 			if (resultValues != null) {
116 				for (Entry<String, Object> values : resultValues.entrySet()) {
117 					if (NumberUtils.isNumeric(values.getValue())) {
118 
119 						logger.info(KeyUtils.getKeyString(server, query, result, values, typeNames, null) + delimiter
120 								+ values.getValue().toString() + delimiter + result.getEpoch());
121 					}
122 				}
123 			}
124 		}
125 	}
126 
127 	/**
128 	 * Initializes the logger. This is called when we need to create a new
129 	 * logger for the given file name.
130 	 *
131 	 * @param fileStr
132 	 * @return a new Logger instance for the given fileStr
133 	 * @throws IOException
134 	 */
135 	protected Logger initLogger(String fileStr) throws IOException {
136 		Appender appender = buildLog4jAppender(fileStr, getMaxLogFileSize(), getMaxLogBackupFiles());
137 		LoggerFactory loggerFactory = buildLog4jLoggerFactory(appender);
138 
139 		String loggerKey = buildLoggerName();
140 
141 		// Create the logger and add to the map of loggers using our factory
142 		LogManager.getLogger(loggerKey, loggerFactory);
143 		return log4jLoggerFactory.getLogger(loggerKey);
144 	}
145 
146 	protected String buildLoggerName() {
147 		return "KeyOutWriter" + this.hashCode();
148 	}
149 
150 	protected Appender buildLog4jAppender(
151 			String fileStr, String maxLogFileSize,
152 			Integer maxLogBackupFiles) throws IOException {
153 
154 		PatternLayout pl = new PatternLayout(LOG_PATTERN);
155 		final RollingFileAppender appender = new RollingFileAppender(pl, fileStr, true);
156 		appender.setImmediateFlush(true);
157 		appender.setBufferedIO(false);
158 		appender.setBufferSize(LOG_IO_BUFFER_SIZE_BYTES);
159 		appender.setMaxFileSize(maxLogFileSize);
160 		appender.setMaxBackupIndex(maxLogBackupFiles);
161 
162 		return appender;
163 	}
164 
165 	protected LoggerFactory buildLog4jLoggerFactory(final Appender appender) {
166 		return new LoggerFactory() {
167 			@Override
168 			public org.apache.log4j.Logger makeNewLoggerInstance(String name) {
169 				org.apache.log4j.Logger logger = org.apache.log4j.Logger.getLogger(name);
170 				logger.addAppender(appender);
171 				logger.setLevel(org.apache.log4j.Level.INFO);
172 				logger.setAdditivity(false);
173 				return logger;
174 			}
175 		};
176 	}

191 		return outputFile;
