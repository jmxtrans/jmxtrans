47  /**
48   * {@link com.googlecode.jmxtrans.model.OutputWriter} for the <a href="https://github.com/OpenTSDB/tcollector/blob/master/collectors/0/udp_bridge.py">TCollector udp_bridge</a>.
49   * Largely based on StatsDWriter and OpenTSDBWriter
50   *
51   * @author Kieren Hynd
52   * @author Arthur Naseef
53   */
54  public class TCollectorUDPWriter extends OpenTSDBGenericWriter {
55  	private static final Logger log = LoggerFactory.getLogger(TCollectorUDPWriter.class);
56  
57  	protected SocketAddress address;
58  	protected DatagramSocket dgSocket;
59  
60  	@JsonCreator
61  	public TCollectorUDPWriter(
62  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
63  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
64  			@JsonProperty("debug") Boolean debugEnabled,
65  			@JsonProperty("host") String host,
66  			@JsonProperty("port") Integer port,
67  			@JsonProperty("tags") Map<String, String> tags,
68  			@JsonProperty("tagName") String tagName,
69  			@JsonProperty("mergeTypeNamesTags") Boolean mergeTypeNamesTags,
70  			@JsonProperty("metricNamingExpression") String metricNamingExpression,
71  			@JsonProperty("addHostnameTag") boolean addHostnameTag,
72  			@JsonProperty("settings") Map<String, Object> settings) throws LifecycleException, UnknownHostException {
73  		super(typeNames, booleanAsNumber, debugEnabled, host, port, tags, tagName, mergeTypeNamesTags, metricNamingExpression,
74  				addHostnameTag, settings);
75  		log.warn("TCollectorUDPWriter is deprecated. Please use TCollectorUDPWriterFactory instead.");
76  	}
77  
78  	/**
79  	 * Do not add the hostname tag "host" with the name of the host by default since tcollector normally adds the
80  	 * hostname.
81  	 */
82  	@Override
83  	protected boolean getAddHostnameTagDefault() {
84  		return false;
85  	}
86  	/**
87  	* Write the results of the query.
88  	 *
89  	 * @param server
90  	 * @param query   - the query and its results.
91  	 * @param results
92  	*/
93  	@Override
94  	public void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
95  		this.startOutput();
96  		for (String resultString : messageFormatter.formatResults(results, server)){
97  			log.debug("TCollectorUDP Message: {}", resultString);
98  			this.sendOutput(resultString);
99  		}
100 		this.finishOutput();
101 	}
102 
103 	/**
104 	 * Setup at start of the writer.
105 	 */
106 	@Override
107 	public void prepareSender() throws LifecycleException {
108 
109 		if (host == null || port == null) {
110 			throw new LifecycleException("Host and port for " + this.getClass().getSimpleName() + " output can't be null");
111 		}
112 
113 		try {
114 			this.dgSocket = new DatagramSocket();
115 			this.address = new InetSocketAddress(host, port);
116 		} catch (SocketException sockExc) {
117 			log.error("Failed to create a datagram socket", sockExc);
118 			throw new LifecycleException(sockExc);
119 		}
120 	}
121 
122 	/**
123 	 * Send a single metric to TCollector.
124 	 *
125 	 * @param metricLine - the line containing the metric name, value, and tags for a single metric; excludes the
126 	 *                   "put" keyword expected by OpenTSDB and the trailing newline character.
127 	 */
128 	@Override
129 	protected void sendOutput(String metricLine) throws IOException {
130 		DatagramPacket packet;
131 		byte[] data;
132 
133 		data = metricLine.getBytes("UTF-8");
134 		packet = new DatagramPacket(data, 0, data.length, this.address);
135 
136 		this.dgSocket.send(packet);
137 	}
