20  /**
21   * A writer for Log4J. It may be a nice way to send JMX metrics to Logstash for example. <br /> <br />
22   * <p/>
23   * Here is an example of MDC variables that are set by this class. <br> <br> server: localhost_9003 <br /> metric:
24   * sun_management_MemoryImpl.HeapMemoryUsage_committed <br /> value: 1251999744 <br /> resultAlias: myHeapMemoryUsage
25   * <br /> attributeName: HeapMemoryUsage <br /> key: committed <br /> epoch: 1388343325728 <br />
26   *
27   * @author Yannick Robin
28   */
29  public class Log4JWriter extends BaseOutputWriter {
30  	private final Logger log;
31  	private final String logger;
32  
33  	@JsonCreator
34  	public Log4JWriter(
35  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
36  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
37  			@JsonProperty("debug") Boolean debugEnabled,
38  			@JsonProperty("logger") String logger,
39  			@JsonProperty("settings") Map<String, Object> settings) {
40  		super(typeNames, booleanAsNumber, debugEnabled, settings);
41  		this.logger = firstNonNull(logger, (String) getSettings().get("logger"), "Log4JWriter");
42  		this.log = Logger.getLogger("Log4JWriter." + this.logger);
43  	}
44  
45  	public void validateSetup(Server server, final Query query) throws ValidationException {}
46  
47  	/**
48  	 * Set the log context and log
49  	 */
50  	public void internalWrite(Server server, final Query query, ImmutableList<Result> results) throws Exception {
51  		final List<String> typeNames = getTypeNames();
52  
53  		for (final Result result : results) {
54  			final Map<String, Object> resultValues = result.getValues();
55  			if (resultValues != null) {
56  				for (final Entry<String, Object> values : resultValues.entrySet()) {
57  					if (NumberUtils.isNumeric(values.getValue())) {
58  						String alias;
59  						if (server.getAlias() != null) {
60  							alias = server.getAlias();
61  						} else {
62  							alias = server.getHost() + "_" + server.getPort();
63  							alias = StringUtils.cleanupStr(alias);
64  						}
65  
66  						MDC.put("server", alias);
67  						MDC.put("metric", KeyUtils.getKeyString(server, query, result, values, typeNames, null));
68  						MDC.put("value", values.getValue());
69  						if (result.getClassNameAlias() != null) {
70  							MDC.put("resultAlias", result.getClassNameAlias());
71  						}
72  						MDC.put("attributeName", result.getAttributeName());
73  						MDC.put("key", values.getKey());
74  						MDC.put("Epoch", String.valueOf(result.getEpoch()));
75  						log.info("");
76  					}
77  				}
78  			}
79  		}
80  	}
81  
82  	public String getLogger() {
83  		return logger;
84  	}
