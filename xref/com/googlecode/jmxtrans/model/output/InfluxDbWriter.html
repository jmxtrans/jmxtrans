53  /**
54   * {@link com.googlecode.jmxtrans.model.OutputWriter} for
55   * <a href="https://influxdb.com/index.html">InfluxDB</a>.
56   *
57   * @author Simon Hutchinson
58   *         <a href="https://github.com/sihutch">github.com/sihutch</a>
59   */
60  @ThreadSafe
61  public class InfluxDbWriter extends OutputWriterAdapter {
62  	private static final Logger log = LoggerFactory.getLogger(InfluxDbWriter.class);
63  
64  	public static final String TAG_HOSTNAME = "hostname";
65  	public static final String JMX_PORT_KEY = "_jmx_port";
66  
67  	@Nonnull private final InfluxDB influxDB;
68  	@Nonnull private final String database;
69  	@Nonnull private final ConsistencyLevel writeConsistency;
70  	@Nonnull private final String retentionPolicy;
71  	@Nonnull private final ImmutableMap<String,String> tags;
72  	@Nonnull ImmutableList<String> typeNames;
73  
74  	/**
75  	 * The {@link ImmutableSet} of {@link ResultAttribute} attributes of
76  	 * {@link Result} that will be written as {@link Point} tags
77  	 */
78  	private final ImmutableSet<ResultAttribute> resultAttributesToWriteAsTags;
79  
80  	private final boolean createDatabase;
81  	private final boolean reportJmxPortAsTag;
82  
83  	private final Predicate<Object> isNotNaN = new Predicate<Object>() {
84  		@Override
85  		public boolean apply(Object input) {
86  			return !input.toString().equals("NaN");
87  		}
88  	};
89  
90  	public InfluxDbWriter(
91  			@Nonnull InfluxDB influxDB,
92  			@Nonnull String database,
93  			@Nonnull ConsistencyLevel writeConsistency,
94  			@Nonnull String retentionPolicy,
95  			@Nonnull ImmutableMap<String,String> tags,
96  			@Nonnull ImmutableSet<ResultAttribute> resultAttributesToWriteAsTags,
97  			@Nonnull ImmutableList<String> typeNames,
98  			boolean createDatabase,
99  			boolean reportJmxPortAsTag) {
100 		this.typeNames = typeNames;
101 		this.database = database;
102 		this.writeConsistency = writeConsistency;
103 		this.retentionPolicy = retentionPolicy;
104 		this.influxDB = influxDB;
105 		this.tags = tags;
106 		this.resultAttributesToWriteAsTags = resultAttributesToWriteAsTags;
107 		this.createDatabase = createDatabase;
108 		this.reportJmxPortAsTag = reportJmxPortAsTag;
109 	}
110 
111 	/**
112 	 * <p>
113 	 * Each {@link Result} is written as a {@link Point} to InfluxDB
114 	 * </p>
115 	 *
116 	 * <p>
117 	 * The measurement for the {@link Point} is to {@link Result#getKeyAlias()}
118 	 * <p>
119 	 * <a href=
120 	 * "https://influxdb.com/docs/v0.9/concepts/key_concepts.html#retention-policy">
121 	 * The retention policy</a> for the measurement is set to "default" unless
122 	 * overridden in settings:
123 	 * </p>
124 	 *
125 	 * <p>
126 	 * The write consistency level defaults to "ALL" unless overridden in
127 	 * settings:
128 	 *
129 	 * <ul>
130 	 * <li>ALL = Write succeeds only if write reached all cluster members.</li>
131 	 * <li>ANY = Write succeeds if write reached any cluster members.</li>
132 	 * <li>ONE = Write succeeds if write reached at least one cluster members.
133 	 * </li>
134 	 * <li>QUORUM = Write succeeds only if write reached a quorum of cluster
135 	 * members.</li>
136 	 * </ul>
137 	 *
138 	 * <p>
139 	 * The time key for the {@link Point} is set to {@link Result#getEpoch()}
140 	 * </p>
141 	 *
142 	 * <p>
143 	 * All {@link Result#getValues()} are written as fields to the {@link Point}
144 	 * </p>
145 	 *
146 	 * <p>
147 	 * The following properties from {@link Result} are written as tags to the
148 	 * {@link Point} unless overriden in settings:
149 	 *
150 	 * <ul>
151 	 * <li>{@link Result#getAttributeName()}</li>
152 	 * <li>{@link Result#getClassName()}</li>
153 	 * <li>{@link Result#getObjDomain()}</li>
154 	 * <li>{@link Result#getTypeName()}</li>
155 	 * </ul>
156 	 * <p>
157 	 * {@link Server#getHost()} is set as a tag on every {@link Point}
158 	 * </p>
159 	 * <p>
160 	 * {@link Server#getPort()} is written as a field, unless {@link #reportJmxPortAsTag} is set to {@code true}
161 	 * </p>
162 	 *
163 	 */
164 	@Override
165 	public void doWrite(Server server, Query query, Iterable<Result> results) throws Exception {
166 		// Creates only if it doesn't already exist
167 		if (createDatabase) influxDB.createDatabase(database);
168 		BatchPoints.Builder batchPointsBuilder = BatchPoints.database(database).retentionPolicy(retentionPolicy)
169 				.tag(TAG_HOSTNAME, server.getSource());
170 
171 		for(Map.Entry<String,String> tag : tags.entrySet()) {
172 			batchPointsBuilder.tag(tag.getKey(),tag.getValue());
173 		}
174 
175 		BatchPoints batchPoints = batchPointsBuilder.consistency(writeConsistency).build();
176 		for (Result result : results) {
177 			log.debug("Query result: {}", result);
178 
179 			HashMap<String, Object> filteredValues = newHashMap();
180 			Object value = result.getValue();
181 			if (isValidNumber(value)) {
182 				String key = KeyUtils.getPrefixedKeyString(query, result, typeNames);
183 				filteredValues.put(key, value);
184 			}
185 
186 			// send the point if filteredValues isn't empty
187 			if (!filteredValues.isEmpty()) {
188 				Map<String, String> resultTagsToApply = buildResultTagMap(result);
189 				if (reportJmxPortAsTag) {
190 					resultTagsToApply.put(JMX_PORT_KEY, server.getPort());
191 				} else {
192 					filteredValues.put(JMX_PORT_KEY, Integer.parseInt(server.getPort()));
193 				}
194 				Point point = Point.measurement(result.getKeyAlias()).time(result.getEpoch(), MILLISECONDS)
195 						.tag(resultTagsToApply).fields(filteredValues).build();
196 
197 				log.debug("Point: {}", point);
198 				batchPoints.point(point);
199 			}
200 		}
201 
202 		influxDB.write(batchPoints);
203 	}
204 
205 	private Map<String, String> buildResultTagMap(Result result) throws Exception {
206 
207 		Map<String, String> resultTagMap = new TreeMap<>();
208 		for (ResultAttribute resultAttribute : resultAttributesToWriteAsTags) {
209 			resultAttribute.addTo(resultTagMap, result);
210 		}
211 
212 		return resultTagMap;
213 
214 	}
