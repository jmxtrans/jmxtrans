38  public class DetailedResultSerializerTest {
39  	@Test
40  	public void convertSingleToStringWhenMultiValues() throws Exception {
41  		ResultSerializer resultSerializer = new DetailedResultSerializer(false);
42  
43  		Collection<String> messages = resultSerializer.serialize(dummyServer(), dummyQuery(), numericResult());
44  
45  		assertThat(messages).hasSize(1);
46  
47  		String message = messages.iterator().next();
48  		// Check JSON Syntax and detailed content
49  		JsonNode jsonNode = new ObjectMapper().readValue(message, JsonNode.class);
50  		assertThat(jsonNode.get("host").asText()).isEqualTo("host.example.net");
51  		assertThat(jsonNode.get("source").asText()).isEqualTo("host.example.net");
52  		assertThat(jsonNode.get("port").asInt()).isEqualTo(4321);
53  		assertThat(jsonNode.get("attributeName").asText()).isEqualTo("ObjectPendingFinalizationCount");
54  		assertThat(jsonNode.get("className").asText()).isEqualTo("sun.management.MemoryImpl");
55  		assertThat(jsonNode.get("objDomain").asText()).isEqualTo("ObjectDomainName");
56  		assertThat(jsonNode.get("typeName").asText()).isEqualTo("type=Memory");
57  		assertThat(jsonNode.get("typeNameMap").get("type").asText()).isEqualTo("Memory");
58  		assertThat(jsonNode.get("typeNameMap")).hasSize(1);
59  		assertThat(jsonNode.get("values").get("ObjectPendingFinalizationCount").asLong()).isEqualTo(10L);
60  		assertThat(jsonNode.get("epoch").asLong()).isEqualTo(0L);
61  		assertThat(jsonNode.get("keyAlias").asText()).isEqualTo("MemoryAlias");
62  	}
