55  /**
56   * Tests for {@link TCollectorUDPWriter}.
57   */
58  @RunWith(PowerMockRunner.class)
59  @PrepareForTest({TCollectorUDPWriter.class, DatagramSocket.class})
60  public class TCollectorUDPWriterTests {
61  	protected TCollectorUDPWriter writer;
62  	protected Query mockQuery;
63  	protected Result mockResult;
64  	protected DatagramSocket mockDgSocket;
65  	protected Logger mockLog;
66  
67  	@Before
68  	public void setupTest() throws Exception {
69  		this.mockQuery = Mockito.mock(Query.class);
70  		this.mockResult = Mockito.mock(Result.class);
71  		this.mockDgSocket = Mockito.mock(DatagramSocket.class);
72  		this.mockLog = Mockito.mock(Logger.class);
73  
74  
75  		// Setup common mock interactions.
76  		PowerMockito.whenNew(DatagramSocket.class).withAnyArguments().thenReturn(this.mockDgSocket);
77  
78  		// When results are needed.
79  		Mockito.when(this.mockResult.getValue()).thenReturn("120021");
80  		Mockito.when(this.mockResult.getValuePath()).thenReturn(ImmutableList.<String>of("x-att1-x"));
81  		Mockito.when(this.mockResult.getAttributeName()).thenReturn("X-ATT-X");
82  		Mockito.when(this.mockResult.getClassName()).thenReturn("X-DOMAIN.PKG.CLASS-X");
83  		Mockito.when(this.mockResult.getTypeName()).thenReturn("Type=x-type-x");
84  
85  
86  		// Prepare the object under test and test data.
87  		Map<String, Object> settings = newHashMap();
88  		settings.put("host", "localhost");
89  		settings.put("port", 8923);
90  
91  		this.writer = TCollectorUDPWriter.builder()
92  				.setDebugEnabled(false)
93  				.setHost("localhost")
94  				.setPort(1234)
95  				.build();
96  
97  		// Inject the mock logger
98  		Whitebox.setInternalState(TCollectorUDPWriter.class, Logger.class, this.mockLog);
99  	}

101 	@Test
102 	public void successfullySendMessageToTCollector() throws Exception {
103 		// Prepare
104 		ArgumentCaptor<DatagramPacket> packetCapture = ArgumentCaptor.forClass(DatagramPacket.class);
105 
106 		// Execute
107 		this.writer.start();
108 		this.writer.doWrite(ServerFixtures.dummyServer(), this.mockQuery, ImmutableList.of(this.mockResult));
109 		this.writer.close();
110 
111 		// Verifications
112 		Mockito.verify(this.mockDgSocket).send(packetCapture.capture());
113 
114 		String sentString = new String(packetCapture.getValue().getData(),
115 				packetCapture.getValue().getOffset(),
116 				packetCapture.getValue().getLength());
117 
118 		Assert.assertThat(sentString, Matchers.startsWith("X-DOMAIN.PKG.CLASS-X.X-ATT-X 0 120021"));
119 		Assert.assertThat(sentString, Matchers.not(Matchers.containsString("host=")));
120 	}

122 	/**
123 	 * Test a socket exception when creating the DatagramSocket.
124 	 */
125 	@Test
126 	public void testSocketException() throws Exception {
127 		// Prepare
128 		SocketException sockExc = new SocketException("X-SOCK-EXC-X");
129 		PowerMockito.whenNew(DatagramSocket.class).withNoArguments().thenThrow(sockExc);
130 
131 		try {
132 			// Execute
133 			this.writer.start();
134 
135 			Assert.fail("LifecycleException missing");
136 		} catch (LifecycleException lcExc) {
137 			// Verify
138 			Assert.assertSame(sockExc, lcExc.getCause());
139 			Mockito.verify(this.mockLog).error(contains("create a datagram socket"), eq(sockExc));
140 		}
141 	}
