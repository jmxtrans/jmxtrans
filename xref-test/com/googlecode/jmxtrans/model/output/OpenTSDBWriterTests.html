55  	/**
56  	 * Prepare the test with standard mocks and mock interactions.  Also perform the base configuration of the
57  	 * object under test.
58  	 */
59  	@Before
60  	public void	setupTest () throws Exception {
61  		// Prepare Mock Objects
62  		this.mockQuery = mock(Query.class);
63  		this.mockResult = mock(Result.class);
64  		this.mockSocket = mock(Socket.class);
65  		this.mockInStreamRdr = mock(InputStreamReader.class);
66  		this.mockBufRdr = mock(BufferedReader.class);
67  		this.mockLog = mock(Logger.class);
68  
69  		// PowerMockito mocks for those final/static classes and methods:
70  		this.mockOut = PowerMockito.mock(DataOutputStream.class);
71  
72  
73  		// Setup common mock interactions.
74  		PowerMockito.whenNew(Socket.class).withParameterTypes(String.class, int.class).
75  			withArguments("localhost", 4242).thenReturn(this.mockSocket);
76  		PowerMockito.whenNew(DataOutputStream.class).withAnyArguments().thenReturn(this.mockOut);
77  		PowerMockito.whenNew(InputStreamReader.class).withAnyArguments().thenReturn(this.mockInStreamRdr);
78  		PowerMockito.whenNew(BufferedReader.class).withAnyArguments().thenReturn(this.mockBufRdr);
79  
80  
81  		// When results are needed.
82  		testValues = ImmutableMap.<String, Object>of("x-att1-x", "120021");
83  		when(this.mockResult.getValues()).thenReturn(testValues);
84  		when(this.mockResult.getAttributeName()).thenReturn("X-ATT-X");
85  		when(this.mockResult.getClassName()).thenReturn("X-DOMAIN.PKG.CLASS-X");
86  		when(this.mockResult.getTypeName()).thenReturn("Type=x-type-x");
87  		when(this.mockInStreamRdr.ready()).thenReturn(false);
88  
89  
90  		// Prepare the object under test and test data.
91  		this.writer = OpenTSDBWriter.builder()
92  				.setDebugEnabled(false)
93  				.setHost("localhost")
94  				.setPort(4242)
95  				.build();
96  
97  		// Inject the mock logger
98  		Whitebox.setInternalState(OpenTSDBWriter.class, Logger.class, this.mockLog);
99  	}

101 	/**
102 	 * Test a successful send without any response from OpenTSDB.
103 	 */
104 	@Test
105 	public void	testSuccessfulSendNoOpenTSDBResponse () throws Exception {
106 		ArgumentCaptor<String>	lineCapture = ArgumentCaptor.forClass(String.class);
107 
108 		// Execute.
109 		this.writer.start();
110 		this.writer.doWrite(null, this.mockQuery, ImmutableList.of(this.mockResult));
111 		this.writer.stop();
112 
113 		// Verify.
114 		verify(this.mockOut).writeBytes(lineCapture.capture());
115 
116 		assertThat(lineCapture.getValue(), Matchers.startsWith("put X-DOMAIN.PKG.CLASS-X.X-ATT-X 0 120021"));
117 		assertThat(lineCapture.getValue(), Matchers.containsString(" host="));
118 	}

167 		UnknownHostException uhExc = new UnknownHostException("X-TEST-UHE-X");
168 		PowerMockito.whenNew(Socket.class).withParameterTypes(String.class, int.class).

170

172 			// Execute.
173 			this.writer.start();
174 
175 			fail("LifecycleException missing");
176 		} catch ( LifecycleException lcExc ) {
177 			// Verify.
178 			assertSame(uhExc, lcExc.getCause());
179 			verify(this.mockLog).error(contains("opening socket"), eq(uhExc));
