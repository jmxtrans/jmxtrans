23  package com.googlecode.jmxtrans.model.output;
24  
25  import com.google.common.collect.ImmutableList;
26  import com.googlecode.jmxtrans.model.Result;
27  import com.googlecode.jmxtrans.model.Server;
28  import com.googlecode.jmxtrans.model.ServerFixtures;
29  import com.googlecode.jmxtrans.model.output.support.opentsdb.OpenTSDBMessageFormatter;
30  import org.junit.Before;
31  import org.junit.Test;
32  import org.mockito.Mockito;
33  
34  import java.io.IOException;
35  import java.io.Writer;
36  import java.util.List;
37  
38  public class TCollectorUDPWriter2Test {
39  
40  	private OpenTSDBMessageFormatter openTSDBMessageFormatter;
41  	private TCollectorUDPWriter2 writer;
42  	private Writer outputWriter;
43  	private Result result;
44  
45  	@Before
46  	public void setup() {
47  		openTSDBMessageFormatter = Mockito.mock(OpenTSDBMessageFormatter.class);
48  		writer = new TCollectorUDPWriter2(openTSDBMessageFormatter);
49  		outputWriter = Mockito.mock(Writer.class);
50  		result = Mockito.mock(Result.class);
51  	}
52  
53  	@Test
54  	public void testMultipleSend() throws IOException {
55  
56  		ImmutableList<Result> results = ImmutableList.of(result, result);
57  		List<String> resultsString = ImmutableList.of("Result1", "Result2");
58  
59  		Mockito.when(openTSDBMessageFormatter.formatResults(results, ServerFixtures.dummyServer())).thenReturn(resultsString);
60  
61  		writer.write(outputWriter, ServerFixtures.dummyServer(), null, results);
62  
63  		Mockito.verify(outputWriter).write("Result1");
64  		Mockito.verify(outputWriter).write("Result2");
65  	}
66  }
