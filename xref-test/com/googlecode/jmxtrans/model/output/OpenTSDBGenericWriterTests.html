400 	protected void validateMergedTypeNameValues(String resultString, boolean mergedInd) {
401 		if (mergedInd) {
402 			assertTrue(resultString.matches("^X-DOMAIN.PKG.CLASS-X\\.X-ATT-X 0 120021.*"));
403 			assertTrue(resultString.matches(".*\\btype=x-att1-x\\b.*"));
404 			assertTrue(resultString.matches(".*\\bTypeGroupNameMissing=x-type-x_x-group-x_x-name-x\\b.*"));
405 		} else {
406 			assertTrue(resultString.matches("^X-DOMAIN.PKG.CLASS-X\\.X-ATT-X 0 120021.*"));
407 			assertTrue(resultString.matches(".*\\btype=x-att1-x\\b.*"));
408 			assertTrue(resultString.matches(".*\\bType=x-type-x\\b.*"));
409 			assertTrue(resultString.matches(".*\\bGroup=x-group-x\\b.*"));
410 			assertTrue(resultString.matches(".*\\bName=x-name-x\\b.*"));
411 			assertTrue(resultString.matches(".*\\bMissing=(\\s.*|$)"));
412 		}
