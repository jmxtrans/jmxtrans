48  	@Test
49  	public void canCreateBasicResultData() throws MalformedObjectNameException, InstanceNotFoundException {
50  		Attribute integerAttribute = new Attribute("StartTime", 51L);
51  		ObjectInstance runtime = getRuntime();
52  		List<Result> results = new JmxResultProcessor(
53  				query,
54  				runtime,
55  				ImmutableList.of(integerAttribute),
56  				runtime.getClassName()).getResults();
57  
58  		assertThat(results).hasSize(1);
59  		Result integerResult = results.get(0);
60  
61  		assertThat(integerResult.getAttributeName()).isEqualTo("StartTime");
62  		assertThat(integerResult.getClassName()).isEqualTo("sun.management.RuntimeImpl");
63  		assertThat(integerResult.getClassNameAlias()).isEqualTo("resultAlias");
64  		assertThat(integerResult.getTypeName()).isEqualTo("type=Runtime");
65  		assertThat(integerResult.getValues()).hasSize(1);
66  	}

114 	@Test
115 	public void canReadTabularData() throws MalformedObjectNameException, AttributeNotFoundException, MBeanException,
116 			ReflectionException, InstanceNotFoundException {
117 		ObjectInstance runtime = getRuntime();
118 		AttributeList attr = ManagementFactory.getPlatformMBeanServer().getAttributes(
119 				runtime.getObjectName(), new String[]{"SystemProperties"});
120 		List<Result> results = new JmxResultProcessor(
121 				query,
122 				runtime,
123 				attr.asList(),
124 				runtime.getClassName()).getResults();
125 
126 		assertThat(results.size()).isGreaterThan(2);
127 
128 		Optional<Result> result = from(results).firstMatch(new ByAttributeName("SystemProperties.java.version"));
129 
130 		assertThat(result.isPresent()).isTrue();
131 
132 		assertThat(result.get().getAttributeName()).isEqualTo("SystemProperties.java.version");
133 		Map<String,Object> values = result.get().getValues();
134 		assertThat(values).hasSize(2);
135 
136 		Object objectValue = result.get().getValues().get("key");
137 		assertThat(objectValue).isInstanceOf(String.class);
138 
139 		String key = (String) objectValue;
140 		assertThat(key).isEqualTo("java.version");
141 	}

179 	private static class ByAttributeName implements Predicate<Result> {
180 		private final String attributeName;
181 
182 		public ByAttributeName(String attributeName) {
183 			this.attributeName = attributeName;
184 		}
185 
186 		@Override
187 		public boolean apply(@Nullable Result result) {
188 			if (result == null) {
189 				return false;
190 			}
191 			return attributeName.equals(result.getAttributeName());
192 		}
