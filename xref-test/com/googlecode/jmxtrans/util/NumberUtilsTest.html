3   import org.junit.Assert;
4   import org.junit.Test;
5   
6   import static com.googlecode.jmxtrans.util.NumberUtils.isNumeric;
7   import static java.lang.Boolean.FALSE;
8   
9   public class NumberUtilsTest {
10  	@Test
11  	public void testIsNumeric() {
12  		Assert.assertFalse(isNumeric(null));
13  		Assert.assertTrue(isNumeric("")); // this is "true" for historical
14  		// reasons
15  		Assert.assertFalse(isNumeric("  "));
16  		Assert.assertTrue(isNumeric("123"));
17  		Assert.assertFalse(isNumeric("12 3"));
18  		Assert.assertFalse(isNumeric("ab2c"));
19  		Assert.assertFalse(isNumeric("12-3"));
20  		Assert.assertTrue(isNumeric("12.3"));
21  		Assert.assertFalse(isNumeric("12.3.3.3"));
22  		Assert.assertTrue(isNumeric(".2"));
23  		Assert.assertFalse(isNumeric("."));
24  		Assert.assertFalse(isNumeric("3."));
25  		Assert.assertTrue(isNumeric(1L));
26  		Assert.assertTrue(isNumeric(2));
27  		Assert.assertTrue(isNumeric((Object) "3.2"));
28  		Assert.assertFalse(isNumeric((Object) "abc"));
29  		Assert.assertFalse(isNumeric(FALSE));
30  	}
31  }
