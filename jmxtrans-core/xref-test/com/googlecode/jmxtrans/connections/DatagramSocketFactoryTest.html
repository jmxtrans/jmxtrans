23  package com.googlecode.jmxtrans.connections;
24  
25  import com.googlecode.jmxtrans.test.RequiresIO;
26  import org.junit.Test;
27  import org.junit.experimental.categories.Category;
28  
29  import java.net.DatagramSocket;
30  import java.net.Inet4Address;
31  import java.net.InetSocketAddress;
32  
33  import static org.assertj.core.api.Assertions.assertThat;
34  
35  @Category(RequiresIO.class)
36  public class DatagramSocketFactoryTest {
37  
38  	public static final int PORT = 50123;
39  
40  	@Test
41  	public void testDatagramSocketFactoryMakeObject() throws Exception {
42  
43  		DatagramSocketFactory socketFactory = new DatagramSocketFactory();
44  
45  		InetSocketAddress socketAddress = new InetSocketAddress(Inet4Address.getLocalHost(), PORT);
46  
47  		DatagramSocket socketObject = socketFactory.makeObject(socketAddress);
48  
49  		// Test if the remote address/port is the correct one.
50  		assertThat(socketObject.getPort()).isEqualTo(PORT);
51  		assertThat(socketObject.getInetAddress()).isEqualTo(Inet4Address.getLocalHost());
52  	}
53  }
