57  /**
58   * Implements the common code for output filters.
59   *
60   * Note that the use of a non threadsafe @link{Map} makes this class non threadsafe.
61   *
62   * @author jon
63   */
64  @NotThreadSafe
65  @ToString
66  public abstract class BaseOutputWriter implements OutputWriter, OutputWriterFactory {
67  
68  	private static final Logger logger = LoggerFactory.getLogger(BaseOutputWriter.class);
69  
70  	public static final String HOST = "host";
71  	public static final String PORT = "port";
72  	public static final String OUTPUT_FILE = "outputFile";
73  	public static final String TEMPLATE_FILE = "templateFile";
74  	public static final String BINARY_PATH = "binaryPath";
75  	public static final String DEBUG = "debug";
76  	public static final String TYPE_NAMES = "typeNames";
77  	public static final String BOOLEAN_AS_NUMBER = "booleanAsNumber";
78  
79  	@Getter private ImmutableList<String> typeNames;
80  	@Getter	private boolean debugEnabled;
81  	private Map<String, Object> settings;
82  	private final ValueTransformer valueTransformer;
83  
84  	@JsonCreator
85  	public BaseOutputWriter(
86  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
87  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
88  			@JsonProperty("debug") Boolean debugEnabled,
89  			@JsonProperty("settings") Map<String, Object> settings) {
90  
91  		if (settings != null && !settings.isEmpty()) {
92  			logger.warn("Using 'settings' is deprecated, please pass attributes directly to the OutputWriter.");
93  		}
94  
95  		// resolve and initialize settings first, so we can refer to them to initialize other fields
96  		this.settings = copyOf(MoreObjects.firstNonNull(
97  				settings,
98  				Collections.<String, Object>emptyMap()));
99  
100 		this.typeNames = copyOf(firstNonNull(
101 				typeNames,
102 				(List<String>) this.settings.get(TYPE_NAMES),
103 				Collections.<String>emptyList()));
104 		this.debugEnabled = firstNonNull(
105 				debugEnabled,
106 				getBooleanSetting(this.settings, DEBUG),
107 				false);
108 
109 		// Get the value of the boolean from the JSON settings if it exists, otherwise default it to the value
110 		// of the boolean passed into the Constructor.
111 		booleanAsNumber = getBooleanSetting(this.settings, BOOLEAN_AS_NUMBER, booleanAsNumber);
112 
113 		if (booleanAsNumber) {
114 			this.valueTransformer = new BooleanAsNumberValueTransformer(1, 0);
115 		} else {
116 			this.valueTransformer = new IdentityValueTransformer();
117 		}
118 	}
119 
120 	protected <T> T firstNonNull(@Nullable T first, @Nullable T second, @Nullable T third) {
121 		return first != null ? first : (second != null ? second : checkNotNull(third));
122 	}
123 	
124 	/**
125 	 * @deprecated Don't use the settings Map, please extract necessary bits at construction time.
126 	 */
127 	@Deprecated @Override
128 	public Map<String, Object> getSettings() {
129 		return settings;
130 	}
131 
132 	/**
133 	 * Given a typeName string, get the first match from the typeNames setting.
134 	 * In other words, suppose you have:
135 	 * <p/>
136 	 * typeName=name=PS Eden Space,type=MemoryPool
137 	 * <p/>
138 	 * If you addTypeName("name"), then it'll retrieve 'PS Eden Space' from the
139 	 * string
140 	 */
141 	protected String getConcatedTypeNameValues(String typeNameStr) {
142 		return TypeNameValuesStringBuilder.getDefaultBuilder().build(this.getTypeNames(), typeNameStr);
143 	}
144 
145 	@Override
146 	public void start() throws LifecycleException {
147 		// Do nothing.
148 	}
149 
150 	@Override
151 	public void close() throws LifecycleException {
152 		// Do nothing.
153 	}
154 
155 	@Override
156 	public final void doWrite(Server server, Query query, Iterable<Result> results) throws Exception {
157 		internalWrite(server, query, from(results).transform(new ResultValuesTransformer(valueTransformer)).toList());
158 	}
159 
160 	protected abstract void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception;
161 
162 	@Override
163 	public OutputWriter create() {
164 		return this;
165 	}
166 
167 }
