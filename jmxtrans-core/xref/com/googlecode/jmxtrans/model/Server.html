216 	private Server(
217 			String alias,
218 			String pid,
219 			String host,
220 			String port,
221 			String username,
222 			String password,
223 			String protocolProviderPackages,
224 			String url,
225 			String cronExpression,
226 			Integer runPeriodSeconds,
227 			Integer numQueryThreads,
228 			boolean local,
229 			boolean ssl,
230 			List<Query> queries,
231 			List<OutputWriterFactory> outputWriterFactories,
232 			List<OutputWriter> outputWriters,
233 			KeyedObjectPool<JmxConnectionProvider, JMXConnection> pool) {
234 
235 		checkArgument(pid != null || url != null || host != null,
236 				"You must provide the pid or the [url|host and port]");
237 		checkArgument(!(pid != null && (url != null || host != null)),
238 				"You must provide the pid OR the url, not both");
239 
240 		this.alias = alias;
241 		this.pid = pid;
242 		this.port = port;
243 		this.username = username;
244 		this.password = password;
245 		this.protocolProviderPackages = protocolProviderPackages;
246 		this.url = url;
247 		this.cronExpression = cronExpression;
248 		if (!isNullOrEmpty(cronExpression)) {
249 			logger.warn("cronExpression is deprecated, please use runPeriodSeconds instead.");
250 		}
251 
252 		this.runPeriodSeconds = runPeriodSeconds;
253 		this.numQueryThreads = firstNonNull(numQueryThreads, 0);
254 		this.local = local;
255 		this.ssl = ssl;
256 		this.queries = copyOf(queries);
257 
258 		// when connecting in local, we cache the host after retrieving it from the network card
259 		if(pid != null) {
260 			try {
261 				this.host = InetAddress.getLocalHost().getHostName();
262 			} catch (UnknownHostException e) {
263 				// should work, so just throw a runtime if it doesn't
264 				throw new RuntimeException(e);
265 			}
266 		}
267 		else {
268 			this.host = host;
269 		}
270 		this.pool = checkNotNull(pool);
271 		this.outputWriterFactories = ImmutableList.copyOf(firstNonNull(outputWriterFactories, ImmutableList.<OutputWriterFactory>of()));
272 		this.outputWriters = ImmutableList.copyOf(firstNonNull(outputWriters, ImmutableList.<OutputWriter>of()));
273 		this.id = String.format("%s_%s_%s", host, port, pid);
274 	}

276 	public Iterable<Result> execute(Query query) throws Exception {
277 		JMXConnection jmxConnection = null;
278 		try {
279 			jmxConnection = pool.borrowObject(this);
280 			ImmutableList.Builder<Result> results = ImmutableList.builder();
281 			MBeanServerConnection connection = jmxConnection.getMBeanServerConnection();
282 
283 			for (ObjectName queryName : query.queryNames(connection)) {
284 				results.addAll(query.fetchResults(connection, queryName));
285 			}
286 
287 			return results.build();
288 		} catch (Exception e) {
289 			if (jmxConnection != null) {
290 				pool.invalidateObject(this, jmxConnection);
291 				jmxConnection = null;
292 			}
293 			throw e;
294 		}
295 		finally {
296 			if (jmxConnection != null) {
297 				pool.returnObject(this, jmxConnection);
298 			}
299 		}
300 	}

302 	/**
303 	 * Generates the proper username/password environment for JMX connections.
304 	 */
305 	@JsonIgnore
306 	public ImmutableMap<String, ?> getEnvironment() {
307 		if (getProtocolProviderPackages() != null && getProtocolProviderPackages().contains("weblogic")) {
308 			ImmutableMap.Builder<String, String> environment = ImmutableMap.builder();
309 			if ((username != null) && (password != null)) {
310 				environment.put(PROTOCOL_PROVIDER_PACKAGES, getProtocolProviderPackages());
311 				environment.put(SECURITY_PRINCIPAL, username);
312 				environment.put(SECURITY_CREDENTIALS, password);
313 			}
314 			return environment.build();
315 		}
316 
317 		ImmutableMap.Builder<String, Object> environment = ImmutableMap.builder();
318 		if ((username != null) && (password != null)) {
319 			String[] credentials = new String[] {
320 					username,
321 					password
322 			};
323 			environment.put(JMXConnector.CREDENTIALS, credentials);
324 		}
325 
326 		JmxTransRMIClientSocketFactory rmiClientSocketFactory = new JmxTransRMIClientSocketFactory(DEFAULT_SOCKET_SO_TIMEOUT_MILLIS, ssl);
327 		// The following is required when JMX is secured with SSL
328 		// with com.sun.management.jmxremote.ssl=true
329 		// as shown in http://docs.oracle.com/javase/8/docs/technotes/guides/management/agent.html#gdfvq
330 		environment.put(RMI_CLIENT_SOCKET_FACTORY_ATTRIBUTE, rmiClientSocketFactory);
331 		// The following is required when JNDI Registry is secured with SSL
332 		// with com.sun.management.jmxremote.registry.ssl=true
333 		// This property is defined in com.sun.jndi.rmi.registry.RegistryContext.SOCKET_FACTORY
334 		environment.put("com.sun.jndi.rmi.factory.socket", rmiClientSocketFactory);
335 
336 		return environment.build();
337 	}

362 	public String getHost() {
363 		if (host == null && url == null) {
364 			return null;
365 		}
366 
367 		if (host != null) {
368 			return host;
369 		}
370 
371 		// removed the caching of the extracted host as it is a very minor
372 		// optimization we should probably pre compute it in the builder and
373 		// throw exception at construction if both url and host are set
374 		// we might also be able to use java.net.URI to parse the URL, but I'm
375 		// not familiar enough with JMX URLs to think of the test cases ...
376 		return url.substring(url.lastIndexOf("//") + 2, url.lastIndexOf(':'));
377 	}

387 	public String getPort() {
388 		if (port == null && url == null) {
389 			return null;
390 		}
391 		if (this.port != null) {
392 			return port;
393 		}
394 
395 		return extractPortFromUrl(url);
396 	}

398 	private static String extractPortFromUrl(String url) {
399 		String computedPort = url.substring(url.lastIndexOf(':') + 1);
400 		if (computedPort.contains("/")) {
401 			computedPort = computedPort.substring(0, computedPort.indexOf('/'));
402 		}
403 		return computedPort;
404 	}

406 	/**
407 	 * The jmx url to connect to. If null, it builds this from host/port with a
408 	 * standard configuration. Other JVM's may want to set this value.
409 	 */
410 	public String getUrl() {
411 		if (this.url == null) {
412 			if ((this.host == null) || (this.port == null)) {
413 				return null;
414 			}
415 			return FRONT + this.host + ":" + this.port + BACK;
416 		}
417 		return this.url;
418 	}

440 	/**
441 	 * Factory to create a JMXServiceURL from a pid. Inner class to prevent class
442 	 * loader issues when tools.jar isn't present.
443 	 */
444 	private static class JMXServiceURLFactory {
445 
446 		private JMXServiceURLFactory() {}
447 
448 		public static JMXServiceURL extractJMXServiceURLFromPid(String pid) throws IOException {
449 
450 			try {
451 				VirtualMachine vm = VirtualMachine.attach(pid);
452 
453 				try {
454 					String connectorAddress = vm.getAgentProperties().getProperty(CONNECTOR_ADDRESS);
455 
456 					if (connectorAddress == null) {
457 						String agent = vm.getSystemProperties().getProperty("java.home") +
458 								File.separator + "lib" + File.separator + "management-agent.jar";
459 						vm.loadAgent(agent);
460 
461 						connectorAddress = vm.getAgentProperties().getProperty(CONNECTOR_ADDRESS);
462 					}
463 
464 					return new JMXServiceURL(connectorAddress);
465 				} finally {
466 					vm.detach();
467 				}
468 			}
469 			catch(Exception e) {
470 				throw new IOException(e);
471 			}
472 		}
473

487 		@Setter private String alias;
488 		@Setter private String pid;
489 		@Setter private String host;
490 		@Setter private String port;
491 		@Setter private String username;
492 		@Setter private String password;
493 		@Setter private String protocolProviderPackages;

495 		@Setter private String cronExpression;
496 		@Setter private Integer runPeriodSeconds;
497 		@Setter private Integer numQueryThreads;
498 		@Setter private boolean local;
499 		@Setter private boolean ssl;
500 		private final List<OutputWriterFactory> outputWriterFactories = new ArrayList<>();
501 		private final List<OutputWriter> outputWriters = new ArrayList<>();
502 		private final List<Query> queries = new ArrayList<>();
503 		@Setter private KeyedObjectPool<JmxConnectionProvider, JMXConnection> pool;
504 
505 		private Builder() {}
506 
507 		private Builder(Server server) {
508 			this.alias = server.alias;
509 			this.pid = server.pid;
510 			this.host = server.pid != null ? null : server.host; // let the host be deduced in the constructor
511 			this.port = server.port;
512 			this.username = server.username;
513 			this.password = server.password;
514 			this.protocolProviderPackages = server.protocolProviderPackages;
515 			this.url = server.url;
516 			this.cronExpression = server.cronExpression;
517 			this.runPeriodSeconds = server.runPeriodSeconds;
518 			this.numQueryThreads = server.numQueryThreads;
519 			this.local = server.local;
520 			this.ssl = server.ssl;
521 			this.queries.addAll(server.queries);
522 			this.pool = server.pool;
523 		}

526 			queries.clear();
527 			return this;
528 		}

533 		}
534 
535 		public Builder addQueries(Set<Query> queries) {
536 			this.queries.addAll(queries);
537 			return this;
538 		}
539 
540 		public Builder addOutputWriterFactory(OutputWriterFactory outputWriterFactory) {
541 			this.outputWriterFactories.add(outputWriterFactory);
542 			return this;
543 		}
544 
545 		public Builder addOutputWriters(Collection<OutputWriter> outputWriters) {
546 			this.outputWriters.addAll(outputWriters);
547 			return this;
548 		}
549

551 			if (!outputWriterFactories.isEmpty()) {
552 				return new Server(
553 						alias,
554 						pid,
555 						host,
556 						port,
557 						username,
558 						password,
559 						protocolProviderPackages,
560 						url,
561 						cronExpression,
562 						runPeriodSeconds,
563 						numQueryThreads,
564 						local,
565 						ssl,
566 						queries,
567 						outputWriterFactories,
568 						pool);
569 			}
570 			return new Server(
571 					alias,
572 					pid,
573 					host,
574 					port,
575 					username,
576 					password,
577 					protocolProviderPackages,
578 					url,
579 					cronExpression,
580 					runPeriodSeconds,
581 					numQueryThreads,
582 					local,
583 					ssl,
584 					queries,
585 					ImmutableList.copyOf(outputWriters),
586 					pool);
587 
588 		}
