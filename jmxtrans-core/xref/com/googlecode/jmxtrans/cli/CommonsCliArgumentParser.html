23  package com.googlecode.jmxtrans.cli;
24  
25  import org.apache.commons.cli.CommandLine;
26  import org.apache.commons.cli.CommandLineParser;
27  import org.apache.commons.cli.GnuParser;
28  import org.apache.commons.cli.HelpFormatter;
29  import org.apache.commons.cli.Option;
30  import org.apache.commons.cli.OptionBuilder;
31  import org.apache.commons.cli.Options;
32  
33  import javax.annotation.Nonnull;
34  import java.io.File;
35  
36  @SuppressWarnings("squid:S1132") // string literals on the right side look just fine to me here
37  public class CommonsCliArgumentParser implements CliArgumentParser {
38  	/** Parse the options given on the command line. */
39  	@Nonnull
40  	@Override
41  	public JmxTransConfiguration parseOptions(@Nonnull String[] args) throws OptionsException, org.apache.commons.cli.ParseException {
42  		CommandLineParser parser = new GnuParser();
43  		CommandLine cl = parser.parse(getOptions(), args);
44  		Option[] options = cl.getOptions();
45  
46  		JmxTransConfiguration configuration = new JmxTransConfiguration();
47  
48  		for (Option option : options) {
49  			if (option.getOpt().equals("c")) {
50  				configuration.setContinueOnJsonError(Boolean.parseBoolean(option.getValue()));
51  			} else if (option.getOpt().equals("j")) {
52  				File jsonDir = new File(option.getValue());
53  				if (jsonDir.exists() && jsonDir.isDirectory()) {
54  					configuration.setProcessConfigDir(jsonDir);
55  				} else {
56  					throw new OptionsException("Path to json directory is invalid: " + jsonDir);
57  				}
58  			} else if (option.getOpt().equals("f")) {
59  				File jsonFile = new File(option.getValue());
60  				if (jsonFile.exists() && jsonFile.isFile()) {
61  					configuration.setProcessConfigFile(jsonFile);
62  				} else {
63  					throw new OptionsException("Path to json file is invalid: " + jsonFile);
64  				}
65  			} else if (option.getOpt().equals("e")) {
66  				configuration.setRunEndlessly(true);
67  			} else if (option.getOpt().equals("q")) {
68  				File quartzConfigFile = new File(option.getValue());
69  				if (quartzConfigFile.exists() && quartzConfigFile.isFile()) {
70  					configuration.setQuartzPropertiesFile(quartzConfigFile);
71  				} else {
72  					throw new OptionsException("Could not find path to the quartz properties file: "
73  							+ quartzConfigFile.getAbsolutePath());
74  				}
75  			} else if (option.getOpt().equals("s")) {
76  				try {
77  					configuration.setRunPeriod(Integer.parseInt(option.getValue()));
78  				} catch (NumberFormatException nfe) {
79  					throw new OptionsException("Seconds between server job runs must be an integer");
80  				}
81  			} else if (option.getOpt().equals("a")) {
82  				configuration.setAdditionalJars(option.getValuesList());
83  			} else if (option.getOpt().equals("h")) {
84  				HelpFormatter formatter = new HelpFormatter();
85  				formatter.printHelp("java -jar jmxtrans-all.jar", getOptions());
86  				configuration.setHelp(true);
87  			}
88  		}
89  		if ((!configuration.isHelp()) && (configuration.getProcessConfigDirOrFile() == null)) {
90  			throw new OptionsException("Please specify either the -f or -j option.");
91  		}
92  		return configuration;
93  	}
94  
95  	@Nonnull
96  	private Options getOptions() {
97  		Options options = new Options();
98  		options.addOption("c", true, "Continue processing even if one of the JSON configuration file is invalid.");
99  		options.addOption("j", true, "Directory where json configuration is stored. Default is .");
100 		options.addOption("f", true, "A single json file to execute.");
101 		options.addOption("e", false, "Run endlessly. Default false.");
102 		options.addOption("q", true, "Path to quartz configuration file.");
103 		options.addOption("s", true, "Seconds between server job runs (not defined with cron). Default: 60");
104 		options.addOption(OptionBuilder
105 				.withArgName("a")
106 				.withLongOpt("additionalJars")
107 				.hasArgs()
108 				.withValueSeparator(',')
109 				.withDescription("Comma delimited list of additional jars to add to the class path")
110 				.create("a"));
111 		options.addOption("h", false, "Help");
112 		return options;
113 	}
114 }
