74  	@Test
75  	public void canCreateBasicResultData() throws MalformedObjectNameException, InstanceNotFoundException {
76  		Attribute integerAttribute = new Attribute("StartTime", 51L);
77  		ObjectInstance runtime = getRuntime();
78  		List<Result> results = new JmxResultProcessor(
79  				query,
80  				runtime,
81  				ImmutableList.of(integerAttribute),
82  				runtime.getClassName(),
83  				TEST_DOMAIN_NAME).getResults();
84  
85  		assertThat(results).hasSize(1);
86  		Result integerResult = results.get(0);
87  
88  		assertThat(integerResult.getAttributeName()).isEqualTo("StartTime");
89  		assertThat(integerResult.getClassName()).isEqualTo("sun.management.RuntimeImpl");
90  		assertThat(integerResult.getKeyAlias()).isEqualTo("resultAlias");
91  		assertThat(integerResult.getTypeName()).isEqualTo("type=Runtime");
92  		assertThat(integerResult.getValues()).hasSize(1);
93  	}

99  		List<Result> results = new JmxResultProcessor(

105 
106 		assertThat(results).hasSize(1);

160 	@Test
161 	public void canReadTabularData() throws MalformedObjectNameException, AttributeNotFoundException, MBeanException,
162 			ReflectionException, InstanceNotFoundException {
163 		ObjectInstance runtime = getRuntime();
164 		AttributeList attr = ManagementFactory.getPlatformMBeanServer().getAttributes(
165 				runtime.getObjectName(), new String[]{"SystemProperties"});
166 		List<Result> results = new JmxResultProcessor(
167 				query,
168 				runtime,
169 				attr.asList(),
170 				runtime.getClassName(),
171 				TEST_DOMAIN_NAME).getResults();
172 
173 		assertThat(results.size()).isGreaterThan(2);
174 
175 		Optional<Result> result = from(results).firstMatch(new ByAttributeName("SystemProperties.java.version"));
176 
177 		assertThat(result.isPresent()).isTrue();
178 
179 		assertThat(result.get().getAttributeName()).isEqualTo("SystemProperties.java.version");
180 		Map<String,Object> values = result.get().getValues();
181 		assertThat(values).hasSize(2);
182 
183 		Object objectValue = result.get().getValues().get("key");
184 		assertThat(objectValue).isInstanceOf(String.class);
185 
186 		String key = (String) objectValue;
187 		assertThat(key).isEqualTo("java.version");
188 	}

265 	private static class ByAttributeName implements Predicate<Result> {
266 		private final String attributeName;
267 
268 		public ByAttributeName(String attributeName) {
269 			this.attributeName = attributeName;
270 		}
271 
272 		@Override
273 		public boolean apply(@Nullable Result result) {
274 			if (result == null) {
275 				return false;
276 			}
277 			return attributeName.equals(result.getAttributeName());
278 		}
