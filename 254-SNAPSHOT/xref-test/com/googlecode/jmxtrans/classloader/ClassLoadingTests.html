23  package com.googlecode.jmxtrans.classloader;
24  
25  import com.googlecode.jmxtrans.test.IntegrationTest;
26  import com.kaching.platform.testing.AllowLocalFileAccess;
27  import org.junit.Ignore;
28  import org.junit.Test;
29  import org.junit.experimental.categories.Category;
30  
31  import java.io.File;
32  import java.io.FileNotFoundException;
33  import java.net.MalformedURLException;
34  import java.net.URISyntaxException;
35  
36  @Category(IntegrationTest.class)
37  @AllowLocalFileAccess(paths = "*")
38  public class ClassLoadingTests {
39  
40  	@Test(expected = ClassNotFoundException.class)
41  	public void classCannotBeAccessedIfJarIsNotLoaded() throws ClassNotFoundException {
42  		Class.forName("dummy.Dummy");
43  	}
44  
45  	/**
46  	 * This test modify the class loader. This makes it hard to isolate, so let's just run it manually.
47  	 */
48  	@Test
49  	@Ignore("Manual test")
50  	public void loadedJarCanBeAccessed() throws ClassNotFoundException, MalformedURLException, FileNotFoundException, URISyntaxException {
51  		File jarFile = new File(ClassLoadingTests.class.getResource("/dummy.jar").toURI());
52  
53  		new ClassLoaderEnricher().add(jarFile);
54  		Class.forName("dummy.Dummy");
55  	}
56  
57  	@Test(expected = FileNotFoundException.class)
58  	public void loadingNonExistingFileThrowsException() throws MalformedURLException, FileNotFoundException {
59  		new ClassLoaderEnricher().add(new File("/nonexising"));
60  	}
61  
62  }
