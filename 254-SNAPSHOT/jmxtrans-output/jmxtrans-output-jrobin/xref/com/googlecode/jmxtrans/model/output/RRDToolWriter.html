60  /**
61   * This takes a JRobin template.xml file and then creates the database if it
62   * doesn't already exist.
63   * 
64   * It will then write the contents of the Query (the Results) to the database.
65   * 
66   * This method exec's out to use the command line version of rrdtool. You need
67   * to specify the path to the directory where the binary rrdtool lives.
68   * 
69   * @author jon
70   */
71  public class RRDToolWriter extends BaseOutputWriter {
72  	private static final Logger log = LoggerFactory.getLogger(RRDToolWriter.class);
73  
74  	public static final String GENERATE = "generate";
75  	private static final char[] INITIALS = { ' ', '.' };
76  
77  	private final File outputFile;
78  	private final File templateFile;
79  	private final File binaryPath;
80  	private final boolean generate;
81  
82  	@JsonCreator
83  	public RRDToolWriter(
84  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
85  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
86  			@JsonProperty("debug") Boolean debugEnabled,
87  			@JsonProperty("outputFile") String outputFile,
88  			@JsonProperty("templateFile") String templateFile,
89  			@JsonProperty("binaryPath") String binaryPath,
90  			@JsonProperty("generate") Boolean generate,
91  			@JsonProperty("settings") Map<String, Object> settings) {
92  		super(typeNames, booleanAsNumber, debugEnabled, settings);
93  		this.outputFile = new File(MoreObjects.firstNonNull(outputFile, (String) getSettings().get(OUTPUT_FILE)));
94  		this.templateFile = new File(MoreObjects.firstNonNull(templateFile, (String) getSettings().get(TEMPLATE_FILE)));
95  		this.binaryPath = new File(MoreObjects.firstNonNull(binaryPath, (String) getSettings().get(BINARY_PATH)));
96  		this.generate = MoreObjects.firstNonNull(generate, Settings.getBooleanSetting(getSettings(), "generate", Boolean.TRUE));
97  
98  		checkState(this.outputFile.exists(), "Output file must exist");
99  		checkState(this.templateFile.exists(), "Template file must exist");
100 		checkState(this.binaryPath.exists(), "RRD Binary must exist");
101 	}

106 	/**
107 	 * rrd datasources must be less than 21 characters in length, so work to
108 	 * make it shorter. Not ideal at all, but works fairly well it seems.
109 	 */
110 	public String getDataSourceName(String typeName, String attributeName, String entry) {
111 		String result;
112 		if (typeName != null) {
113 			result = typeName + attributeName + entry;
114 		} else {
115 			result = attributeName + entry;
116 		}
117 
118 		if (attributeName.length() > 15) {
119 			String[] split = StringUtils.splitByCharacterTypeCamelCase(attributeName);
120 			String join = StringUtils.join(split, '.');
121 			attributeName = WordUtils.initials(join, INITIALS);
122 		}
123 		result = attributeName + DigestUtils.md5Hex(result);
124 
125 		result = StringUtils.left(result, 19);
126 
127 		return result;
128 	}

130 	@Override
131 	public void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
132 		RrdDef def = getDatabaseTemplateSpec();
133 
134 		List<String> dsNames = getDsNames(def.getDsDefs());
135 
136 		Map<String, String> dataMap = new TreeMap<String, String>();
137 
138 		// go over all the results and look for datasource names that map to
139 		// keys from the result values
140 		for (Result res : results) {
141 			log.debug(res.toString());
142 			Map<String, Object> values = res.getValues();
143 			if (values != null) {
144 				for (Entry<String, Object> entry : values.entrySet()) {
145 					String key = getDataSourceName(getConcatedTypeNameValues(res.getTypeName()), res.getAttributeName(), entry.getKey());
146 
147 					if (isNumeric(entry.getValue())) {
148 						log.debug("Generated DataSource name:value: {} : {}", key, entry.getValue());
149 						if (dsNames.contains(key)) {
150 							dataMap.put(key, entry.getValue().toString());
151 						}
152 					}
153 				}
154 			}
155 		}
156 
157 		doGenerate(results);
158 
159 		if (!dataMap.keySet().isEmpty() && !dataMap.values().isEmpty()) {
160 			rrdToolUpdate(StringUtils.join(dataMap.keySet(), ':'), StringUtils.join(dataMap.values(), ':'));
161 		} else {
162 			log.error("Nothing was logged for query: " + query);
163 		}
164 	}

166 	private void doGenerate(List<Result> results) throws Exception {
167 		if (isDebugEnabled() && generate) {
168 			StringBuilder sb = new StringBuilder("\n");
169 			List<String> keys = new ArrayList<String>();
170 
171 			for (Result res : results) {
172 				Map<String, Object> values = res.getValues();
173 				if (values != null) {
174 					for (Entry<String, Object> entry : values.entrySet()) {
175 						if (isNumeric(entry.getValue())) {
176 							String key = getDataSourceName(getConcatedTypeNameValues(res.getTypeName()), res.getAttributeName(), entry.getKey());
177 							if (keys.contains(key)) {
178 								throw new Exception("Duplicate datasource name found: '" + key
179 										+ "'. Please try to add more typeName keys to the writer to make the name more unique. " + res.toString());
180 							}
181 							keys.add(key);
182 
183 							sb.append("<datasource><!-- ").append(res.getTypeName()).append(":")
184 									.append(res.getAttributeName()).append(":").append(entry.getKey())
185 									.append(" --><name>").append(key)
186 									.append("</name><type>GAUGE</type><heartbeat>400</heartbeat><min>U</min><max>U</max></datasource>\n");
187 						}
188 					}
189 				}
190 			}
191 			log.debug(sb.toString());
192 		}
193 	}

195 	/**
196 	 * Executes the rrdtool update command.
197 	 */
198 	protected void rrdToolUpdate(String template, String data) throws Exception {
199 		List<String> commands = new ArrayList<String>();
200 		commands.add(binaryPath + "/rrdtool");
201 		commands.add("update");
202 		commands.add(outputFile.getCanonicalPath());
203 		commands.add("-t");
204 		commands.add(template);
205 		commands.add("N:" + data);
206 
207 		ProcessBuilder pb = new ProcessBuilder(commands);
208 		Process process = pb.start();
209 		checkErrorStream(process);
210 	}

212 	/**
213 	 * If the database file doesn't exist, it'll get created, otherwise, it'll
214 	 * be returned in r/w mode.
215 	 */
216 	protected RrdDef getDatabaseTemplateSpec() throws Exception {
217 		RrdDefTemplate t = new RrdDefTemplate(templateFile);
218 		t.setVariable("database", this.outputFile.getCanonicalPath());
219 		RrdDef def = t.getRrdDef();
220 		if (!this.outputFile.exists()) {
221 			FileUtils.forceMkdir(this.outputFile.getParentFile());
222 			rrdToolCreateDatabase(def);
223 		}
224 		return def;
225 	}

227 	/**
228 	 * Calls out to the rrdtool binary with the 'create' command.
229 	 */
230 	protected void rrdToolCreateDatabase(RrdDef def) throws Exception {
231 		List<String> commands = new ArrayList<String>();
232 		commands.add(this.binaryPath + "/rrdtool");
233 		commands.add("create");
234 		commands.add(this.outputFile.getCanonicalPath());
235 		commands.add("-s");
236 		commands.add(String.valueOf(def.getStep()));
237 
238 		for (DsDef dsdef : def.getDsDefs()) {
239 			commands.add(getDsDefStr(dsdef));
240 		}
241 
242 		for (ArcDef adef : def.getArcDefs()) {
243 			commands.add(getRraStr(adef));
244 		}
245 
246 		ProcessBuilder pb = new ProcessBuilder(commands);
247 		Process process = pb.start();
248 		try {
249 			checkErrorStream(process);
250 		} finally {
251 			IOUtils.closeQuietly(process.getInputStream());
252 			IOUtils.closeQuietly(process.getOutputStream());
253 			IOUtils.closeQuietly(process.getErrorStream());
254 		}
255 	}

257 	/**
258 	 * Check to see if there was an error processing an rrdtool command
259 	 */
260 	private void checkErrorStream(Process process) throws Exception {
261 		Closer closer = Closer.create();
262 		try {
263 			InputStream is = closer.register(process.getErrorStream());
264 			// rrdtool should use platform encoding (unless you did something
265 			// very strange with your installation of rrdtool). So let's be
266 			// explicit and use the presumed correct encoding to read errors.
267 			InputStreamReader isr = closer.register(new InputStreamReader(is, Charset.defaultCharset()));
268 			BufferedReader br = closer.register(new BufferedReader(isr));
269 			StringBuilder sb = new StringBuilder();
270 			String line;
271 			while ((line = br.readLine()) != null) {
272 				sb.append(line);
273 			}
274 			if (sb.length() > 0) {
275 				throw new RuntimeException(sb.toString());
276 			}
277 		} catch (Throwable t) {
278 			throw closer.rethrow(t);
279 		} finally {
280 			closer.close();
281 		}
282 	}

284 	/**
285 	 * Generate a RRA line for rrdtool
286 	 */
287 	private String getRraStr(ArcDef def) {
288 		return "RRA:" + def.getConsolFun() + ":" + def.getXff() + ":" + def.getSteps() + ":" + def.getRows();
289 	}

291 	/**
292 	 * "rrdtool create temperature.rrd --step 300 \\\n" +
293 	 * "         DS:temp:GAUGE:600:-273:5000 \\\n" +
294 	 * "         RRA:AVERAGE:0.5:1:1200 \\\n" +
295 	 * "         RRA:MIN:0.5:12:2400 \\\n" + "         RRA:MAX:0.5:12:2400 \\\n"
296 	 * + "         RRA:AVERAGE:0.5:12:2400"
297 	 */
298 	private String getDsDefStr(DsDef def) {
299 		return "DS:" + def.getDsName() + ":" + def.getDsType() + ":" + def.getHeartbeat() + ":" + formatDouble(def.getMinValue()) + ":"
300 				+ formatDouble(def.getMaxValue());
301 	}

303 	/**
304 	 * Get a list of DsNames used to create the datasource.
305 	 */
306 	private List<String> getDsNames(DsDef[] defs) {
307 		List<String> names = new ArrayList<String>();
308 		for (DsDef def : defs) {
309 			names.add(def.getDsName());
310 		}
311 		return names;
312 	}

314 	/**
315 	 * If dbl is NaN, then return U
316 	 */
317 	private String formatDouble(double dbl) {
318 		if (Double.isNaN(dbl)) {
319 			return "U";
320 		}
321 		return String.valueOf(dbl);
322 	}
