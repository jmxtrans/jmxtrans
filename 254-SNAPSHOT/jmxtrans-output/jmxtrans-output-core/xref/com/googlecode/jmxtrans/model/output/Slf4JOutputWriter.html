26  import com.fasterxml.jackson.annotation.JsonProperty;
27  import com.google.common.annotations.VisibleForTesting;
28  import com.google.common.base.MoreObjects;
29  import com.google.common.collect.ImmutableList;
30  import com.google.common.io.Closer;
31  import com.googlecode.jmxtrans.model.Query;
32  import com.googlecode.jmxtrans.model.Result;
33  import com.googlecode.jmxtrans.model.Server;
34  import com.googlecode.jmxtrans.model.ValidationException;
35  import com.googlecode.jmxtrans.model.naming.KeyUtils;
36  import lombok.EqualsAndHashCode;
37  import lombok.ToString;
38  import org.slf4j.Logger;
39  import org.slf4j.LoggerFactory;
40  import org.slf4j.MDC;
41  
42  import java.io.IOException;
43  import java.util.HashMap;
44  import java.util.List;
45  import java.util.Map;
46  
47  import static com.googlecode.jmxtrans.model.naming.StringUtils.cleanupStr;
48  import static com.googlecode.jmxtrans.util.NumberUtils.isNumeric;
49  import static java.lang.String.valueOf;
50  
51  @EqualsAndHashCode
52  @ToString
53  public class Slf4JOutputWriter extends BaseOutputWriter {
54  
55  	private final Logger logger;
56  
57  	@JsonCreator
58  	public Slf4JOutputWriter(
59  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
60  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
61  			@JsonProperty("debug") Boolean debugEnabled,
62  			@JsonProperty("logger") String logger,
63  			@JsonProperty("settings") Map<String, Object> settings) {
64  		super(typeNames, booleanAsNumber, debugEnabled, settings);
65  		String loggerName = MoreObjects.firstNonNull(logger, "jmxtrans.output");
66  		this.logger = LoggerFactory.getLogger(loggerName);
67  	}
68  
69  	@VisibleForTesting
70  	Slf4JOutputWriter(Logger logger) {
71  		super(ImmutableList.<String>of(),
72  				true,
73  				false,
74  				new HashMap<String, Object>());
75  		this.logger = logger;
76  	}
77  
78  	@Override
79  	protected void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
80  		final List<String> typeNames = getTypeNames();
81  
82  		for (final Result result : results) {
83  			for (final Map.Entry<String, Object> values : result.getValues().entrySet()) {
84  				logValue(server, query, typeNames, result, values);
85  			}
86  		}
87  	}
88  
89  	private void logValue(Server server, Query query, List<String> typeNames, Result result, Map.Entry<String, Object> values) throws IOException {
90  		Object value = values.getValue();
91  
92  		if (value != null && isNumeric(value)) {
93  			Closer closer = Closer.create();
94  			try {
95  				closer.register(MDC.putCloseable("server", computeAlias(server)));
96  				closer.register(MDC.putCloseable("metric", KeyUtils.getKeyString(server, query, result, values, typeNames, null)));
97  				closer.register(MDC.putCloseable("value", value.toString()));
98  				if (result.getKeyAlias() != null) {
99  					closer.register(MDC.putCloseable("resultAlias", result.getKeyAlias()));
100 				}
101 				closer.register(MDC.putCloseable("attributeName", result.getAttributeName()));
102 				closer.register(MDC.putCloseable("key", values.getKey()));
103 				closer.register(MDC.putCloseable("epoch", valueOf(result.getEpoch())));
104 
105 				logger.info("");
106 			} catch (Throwable t) {
107 				closer.rethrow(t);
108 			} finally {
109 				closer.close();
110 			}
111 		}
112 	}
113 
114 	private String computeAlias(Server server) {
115 		if (server.getAlias() != null) return server.getAlias();
116 		return cleanupStr(server.getHost() + "_" + server.getPort());
117 	}
118 
119 	@Override
120 	public void validateSetup(Server server, Query query) throws ValidationException {}
121 
122 }
