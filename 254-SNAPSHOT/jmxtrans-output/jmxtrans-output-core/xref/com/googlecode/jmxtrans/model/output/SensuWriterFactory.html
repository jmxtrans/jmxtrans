23  package com.googlecode.jmxtrans.model.output;
24  
25  import com.fasterxml.jackson.annotation.JsonProperty;
26  import com.fasterxml.jackson.core.JsonFactory;
27  import com.google.common.collect.ImmutableList;
28  import com.googlecode.jmxtrans.model.OutputWriterFactory;
29  import com.googlecode.jmxtrans.model.output.support.ResultTransformerOutputWriter;
30  import com.googlecode.jmxtrans.model.output.support.TcpOutputWriterBuilder;
31  import com.googlecode.jmxtrans.model.output.support.WriterPoolOutputWriter;
32  import lombok.EqualsAndHashCode;
33  import lombok.ToString;
34  
35  import javax.annotation.Nonnull;
36  import javax.annotation.Nullable;
37  import java.net.InetSocketAddress;
38  
39  import static com.google.common.base.MoreObjects.firstNonNull;
40  
41  @EqualsAndHashCode
42  @ToString
43  public class SensuWriterFactory implements OutputWriterFactory {
44  
45  	private final boolean booleanAsNumber;
46  	@Nonnull private final InetSocketAddress server;
47  	@Nonnull private final ImmutableList<String> typeNames;
48  	@Nullable private final String rootPrefix;
49  
50  	public SensuWriterFactory(
51  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
52  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
53  			@JsonProperty("host") String host,
54  			@JsonProperty("port") Integer port,
55  			@JsonProperty("rootPrefix") String rootPrefix) {
56  		this.rootPrefix = rootPrefix;
57  		this.typeNames = firstNonNull(typeNames, ImmutableList.<String>of());
58  		this.booleanAsNumber = booleanAsNumber;
59  		this.server = new InetSocketAddress(
60  				firstNonNull(host, "localhost"),
61  				firstNonNull(port, 3030));
62  	}
63  
64  	@Override
65  	public ResultTransformerOutputWriter<WriterPoolOutputWriter<SensuWriter2>> create() {
66  		return ResultTransformerOutputWriter.booleanToNumber(
67  				booleanAsNumber,
68  				TcpOutputWriterBuilder.builder(
69  						server,
70  						new SensuWriter2(
71  								new GraphiteWriter2(typeNames, rootPrefix),
72  								new JsonFactory()))
73  						.build());
74  	}
75  
76  }
