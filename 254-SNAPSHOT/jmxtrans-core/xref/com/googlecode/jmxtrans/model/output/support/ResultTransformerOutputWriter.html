23  package com.googlecode.jmxtrans.model.output.support;
24  
25  import com.googlecode.jmxtrans.exceptions.LifecycleException;
26  import com.googlecode.jmxtrans.model.OutputWriter;
27  import com.googlecode.jmxtrans.model.Query;
28  import com.googlecode.jmxtrans.model.Result;
29  import com.googlecode.jmxtrans.model.Server;
30  import com.googlecode.jmxtrans.model.ValidationException;
31  import com.googlecode.jmxtrans.model.results.BooleanAsNumberValueTransformer;
32  import com.googlecode.jmxtrans.model.results.IdentityValueTransformer;
33  import com.googlecode.jmxtrans.model.results.ResultValuesTransformer;
34  
35  import javax.annotation.Nonnull;
36  import java.util.Collections;
37  import java.util.Map;
38  
39  import static com.google.common.collect.FluentIterable.from;
40  
41  public class ResultTransformerOutputWriter<T extends OutputWriter> implements OutputWriter {
42  
43  	@Nonnull private final ResultValuesTransformer resultValuesTransformer;
44  	@Nonnull private final T target;
45  
46  	public ResultTransformerOutputWriter(@Nonnull ResultValuesTransformer resultValuesTransformer, @Nonnull T target) {
47  		this.resultValuesTransformer = resultValuesTransformer;
48  		this.target = target;
49  	}
50  
51  	@Override
52  	public void start() throws LifecycleException {
53  	}
54  
55  	@Override
56  	public void stop() throws LifecycleException {
57  	}
58  
59  	@Override
60  	public void doWrite(Server server, Query query, Iterable<Result> results) throws Exception {
61  		target.doWrite(
62  				server,
63  				query,
64  				from(results).transform(resultValuesTransformer).toList());
65  	}
66  
67  	@Override
68  	public Map<String, Object> getSettings() {
69  		return Collections.emptyMap();
70  	}
71  
72  	@Override
73  	public void validateSetup(Server server, Query query) throws ValidationException {
74  	}
75  
76  	public static <T extends OutputWriter> ResultTransformerOutputWriter<T> booleanToNumber(boolean booleanToNumber, T target) {
77  		if (booleanToNumber) return booleanToNumber(target);
78  		return identity(target);
79  	}
80  
81  	public static <T extends OutputWriter> ResultTransformerOutputWriter<T> booleanToNumber(T target) {
82  		return new ResultTransformerOutputWriter<T>(new ResultValuesTransformer(new BooleanAsNumberValueTransformer(1, 0)), target);
83  	}
84  
85  	public static <T extends OutputWriter> ResultTransformerOutputWriter<T> identity(T target) {
86  		return new ResultTransformerOutputWriter<T>(new ResultValuesTransformer(new IdentityValueTransformer()), target);
87  	}
88  
89  
90  }
