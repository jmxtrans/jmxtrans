23  package com.googlecode.jmxtrans.model.naming.typename;
24  
25  import org.junit.Test;
26  
27  import java.util.Collections;
28  
29  import static java.util.Arrays.asList;
30  import static org.assertj.core.api.Assertions.assertThat;
31  
32  public class PrependingTypeNameValuesStringBuilderTest {
33  
34  	@Test
35  	public void testBuilds() {
36  		PrependingTypeNameValuesStringBuilder builder = new PrependingTypeNameValuesStringBuilder(
37  				TypeNameValuesStringBuilder.DEFAULT_SEPARATOR, asList("p-key1-p", "q-key1-q"));
38  		assertThat(builder.build(null, "p-key1-p=p-value1-p,q-key1-q=q-value1-q")).isEqualTo("p-value1-p_q-value1-q");
39  		assertThat(builder.build(Collections.<String>emptyList(), "p-key1-p=p-value1-p")).isEqualTo("p-value1-p");
40  		assertThat(builder.build(asList("x-key1-x"), "")).isEmpty();
41  		assertThat(builder.build(asList("x-key1-x"), "p-key1-p=p-value1-p")).isEqualTo("p-value1-p");
42  		assertThat(builder.build(asList("x-key1-x", "y-key2-y"), "")).isEmpty();
43  		assertThat(builder.build(asList("x-key1-x"), "x-key1-x=x-value1-x")).isEqualTo("x-value1-x");
44  		assertThat(builder.build(asList("x-key1-x"), "x-key1-x=x-value1-x,q-key1-q=q-value1-q"))
45  				.isEqualTo("q-value1-q_x-value1-x");
46  		assertThat(builder.build(asList("x-key1-x"), "y-key2-y=y-value2-y")).isEmpty();
47  		assertThat(builder.build(asList("x-key1-x", "y-key2-y"), "x-key1-x=x-value1-x,y-key2-y=y-value2-y"))
48  				.isEqualTo("x-value1-x_y-value2-y");
49  		assertThat(builder.build(
50  					asList("x-key1-x", "y-key2-y"),
51  					"x-key1-x=x-value1-x,q-key1-q=q-value1-q,y-key2-y=y-value2-y,p-key1-p=p-value1-p"))
52  				.isEqualTo("p-value1-p_q-value1-q_x-value1-x_y-value2-y");
53  	}
54  }
