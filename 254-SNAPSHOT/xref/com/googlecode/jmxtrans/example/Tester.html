23  package com.googlecode.jmxtrans.example;
24  
25  import com.google.common.collect.ImmutableList;
26  import com.google.inject.Injector;
27  import com.googlecode.jmxtrans.JmxTransformer;
28  import com.googlecode.jmxtrans.cli.JmxTransConfiguration;
29  import com.googlecode.jmxtrans.guice.JmxTransModule;
30  import com.googlecode.jmxtrans.model.JmxProcess;
31  import com.googlecode.jmxtrans.model.Query;
32  import com.googlecode.jmxtrans.model.Server;
33  import com.googlecode.jmxtrans.model.output.StdOutWriter;
34  
35  import java.util.Collections;
36  
37  /**
38   * This class produces the json that is in example.json.
39   * 
40   * @author jon
41   */
42  public class Tester {
43  
44  	private static final JsonPrinter printer = new JsonPrinter(System.out);
45  
46  	/** */
47  	public static void main(String[] args) throws Exception {
48  		Server server = Server.builder()
49  				.setHost("w2")
50  				.setPort("1099")
51  				.setNumQueryThreads(2)
52  				.addQuery(Query.builder()
53  						.setObj("java.lang:type=Memory")
54  						.addAttr("HeapMemoryUsage", "NonHeapMemoryUsage")
55  						.addOutputWriter(new StdOutWriter(ImmutableList.<String>of(), false, false, Collections.<String, Object>emptyMap()))
56  						.build())
57  				.addQuery(Query.builder()
58  						.setObj("java.lang:name=CMS Old Gen,type=MemoryPool")
59  						.addAttr("Usage")
60  						.addOutputWriter(new StdOutWriter(ImmutableList.<String>of(), false, false, Collections.<String, Object>emptyMap()))
61  						.build())
62  				.addQuery(Query.builder()
63  						.setObj("java.lang:name=ConcurrentMarkSweep,type=GarbageCollector")
64  						.addAttr("LastGcInfo")
65  						.addOutputWriter(new StdOutWriter(ImmutableList.<String>of(), false, false, Collections.<String, Object>emptyMap()))
66  						.build())
67  				.build();
68  
69  		JmxProcess process = new JmxProcess(server);
70  		printer.prettyPrint(process);
71  
72  		Injector injector = JmxTransModule.createInjector(new JmxTransConfiguration());
73  		JmxTransformer transformer = injector.getInstance(JmxTransformer.class);
74  
75  		transformer.executeStandalone(process);
76  	}
77  
78  }
