58  /**
59   * This low latency and thread safe output writer sends data to a kafka topics in JSON format.
60   * Kafka Topics can be passed as separated by commas such as kafka01,kafka02.
61   *
62   * @author : utkarsh bhatnagar
63   * @github user : utkarshcmu
64   * @email : utkarsh.cmu@gmail.com
65   */
66  
67  @NotThreadSafe
68  public class KafkaWriter extends BaseOutputWriter {
69  
70  	private static final Logger log = LoggerFactory.getLogger(KafkaWriter.class);
71  
72  	private static final String DEFAULT_ROOT_PREFIX = "servers";
73  	private final JsonFactory jsonFactory;
74  
75  	private Producer<String,String> producer;
76  	private final Iterable<String> topics;
77  	private final String rootPrefix;
78  	private final ImmutableMap<String, String> tags;
79  
80  	@JsonCreator
81  	public KafkaWriter(
82  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
83  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
84  			@JsonProperty("rootPrefix") String rootPrefix,
85  			@JsonProperty("debug") Boolean debugEnabled,
86  			@JsonProperty("topics") String topics,
87  			@JsonProperty("tags") Map<String, String> tags,
88  			@JsonProperty("settings") Map<String, Object> settings) {
89  		super(typeNames, booleanAsNumber, debugEnabled, settings);
90  		this.rootPrefix = firstNonNull(
91  						rootPrefix,
92  						(String) getSettings().get("rootPrefix"),
93  						DEFAULT_ROOT_PREFIX);
94  		// Setting all the required Kafka Properties
95  		Properties kafkaProperties =  new Properties();
96  		kafkaProperties.setProperty("metadata.broker.list", Settings.getStringSetting(settings, "metadata.broker.list", null));
97  		kafkaProperties.setProperty("zk.connect", Settings.getStringSetting(settings, "zk.connect", null));
98  		kafkaProperties.setProperty("serializer.class", Settings.getStringSetting(settings, "serializer.class", null));
99  		this.producer= new Producer<String,String>(new ProducerConfig(kafkaProperties));
100 		this.topics = asList(Settings.getStringSetting(settings, "topics", "").split(","));
101 		this.tags = ImmutableMap.copyOf(firstNonNull(tags, (Map<String, String>) getSettings().get("tags"), ImmutableMap.<String, String>of()));
102 		jsonFactory = new JsonFactory();
103 	}
104 
105 	@Override
106 	public void validateSetup(Server server, Query query) throws ValidationException {}
107 
108 	@Override
109 	protected void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
110 		List<String> typeNames = this.getTypeNames();
111 
112 		for (Result result : results) {
113 			log.debug("Query result: [{}]", result);
114 			Map<String, Object> resultValues = result.getValues();
115 			for (Entry<String, Object> values : resultValues.entrySet()) {
116 				Object value = values.getValue();
117 				if (isNumeric(value)) {
118 					String message = createJsonMessage(server, query, typeNames, result, values, value);
119 					for(String topic : this.topics) {
120 						log.debug("Topic: [{}] ; Kafka Message: [{}]", topic, message);
121 						producer.send(new KeyedMessage<String, String>(topic, message));
122 					}
123 				} else {
124 					log.warn("Unable to submit non-numeric value to Kafka: [{}] from result [{}]", value, result);
125 				}
126 			}
127 		}
128 	}
129 
130 	private String createJsonMessage(Server server, Query query, List<String> typeNames, Result result, Entry<String, Object> values, Object value) throws IOException {
131 		String keyString = getKeyString(server, query, result, values, typeNames, this.rootPrefix);
132 		String cleanKeyString = keyString.replaceAll("[()]", "_");
133 
134 		Closer closer = Closer.create();
135 		try {
136 			ByteArrayOutputStream out = closer.register(new ByteArrayOutputStream());
137 			JsonGenerator generator = closer.register(jsonFactory.createGenerator(out, UTF8));
138 			generator.writeStartObject();
139 			generator.writeStringField("keyspace", cleanKeyString);
140 			generator.writeStringField("value", value.toString());
141 			generator.writeNumberField("timestamp", result.getEpoch() / 1000);
142 			generator.writeObjectFieldStart("tags");
143 
144 			for (Entry<String, String> tag : this.tags.entrySet()) {
145 				generator.writeStringField(tag.getKey(), tag.getValue());
146 			}
147 
148 			generator.writeEndObject();
149 			generator.writeEndObject();
150 			generator.close();
151 			return out.toString("UTF-8");
152 		} catch (Throwable t) {
153 			throw closer.rethrow(t);
154 		} finally {
155 			closer.close();
156 		}
157 	}
158 
159 	@VisibleForTesting
160 	void setProducer(Producer<String, String> producer) {
161 		this.producer = producer;
162 	}
