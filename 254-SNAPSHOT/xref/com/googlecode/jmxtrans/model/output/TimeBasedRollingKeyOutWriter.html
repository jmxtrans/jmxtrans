23  package com.googlecode.jmxtrans.model.output;
24  
25  import ch.qos.logback.classic.Level;
26  import ch.qos.logback.classic.LoggerContext;
27  import ch.qos.logback.classic.encoder.PatternLayoutEncoder;
28  import ch.qos.logback.core.Appender;
29  import ch.qos.logback.core.encoder.Encoder;
30  import ch.qos.logback.core.rolling.RollingFileAppender;
31  import ch.qos.logback.core.rolling.RollingPolicy;
32  import ch.qos.logback.core.rolling.SizeAndTimeBasedFNATP;
33  import ch.qos.logback.core.rolling.TimeBasedRollingPolicy;
34  import com.fasterxml.jackson.annotation.JsonCreator;
35  import com.fasterxml.jackson.annotation.JsonProperty;
36  import com.google.common.collect.ImmutableList;
37  import org.slf4j.Logger;
38  
39  import java.io.IOException;
40  import java.util.Map;
41  
42  /**
43   * Extension of KeyOutWriter to use Logback for its logging.  This version supports outputting with a time and size
44   * based rollover policy but with no file rename taking place.  The filename remains constant for the output file from
45   * its creation to the roll over to a new file.  Then the old file remains with its old name and the new file is used
46   * for further logging.
47   * 
48   * We found this to be useful when dealing with forwarding logs on Windows as our other support applications were not
49   * able to handle the rename correctly on Windows and DailyKeyOutWriter caused the original file to be renamed.  
50   * Also DailyRollingAppender has bugs that can cause data loss and it is no longer recommended for use judging by the
51   * javadocs for it.
52   * 
53   * The basic operation is to set the date time to rollover in the outputFile setting.  For example:
54   *      "outputFile": "testing-here.log%d{yyyy-MM-dd}.%i",
55   * In this case we have set the logging to roll over every day.  There is also a default max size as per the
56   * KeyOutWriter.  If the file exceeds the default max the .%i integer is increments.  See the Logstash documentation
57   * for more information.
58   */
59  public class TimeBasedRollingKeyOutWriter extends KeyOutWriter {
60  	
61  	private static final LoggerContext loggerContext = new LoggerContext();
62  	private static final String DEFAULT_OUTPUT_PATTERN = "%msg%n";
63  	private static final String SETTING_OUTPUT_PATTERN = "outputPattern";
64  
65  	@JsonCreator
66  	public TimeBasedRollingKeyOutWriter(
67  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
68  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
69  			@JsonProperty("debug") Boolean debugEnabled,
70  			@JsonProperty("outputFile") String outputFile,
71  			@JsonProperty("maxLogFileSize") String maxLogFileSize,
72  			@JsonProperty("maxLogBackupFiles") int maxLogBackupFiles,
73  			@JsonProperty("delimiter") String delimiter,
74  			@JsonProperty("settings") Map<String, Object> settings) {
75  		super(typeNames, booleanAsNumber, debugEnabled, outputFile, maxLogFileSize, maxLogBackupFiles, delimiter, settings);
76  	}
77  
78  	@Override
79  	protected Logger initLogger(String fileStr) throws IOException {
80  		RollingPolicy rollingPolicy = initRollingPolicy(fileStr, getMaxLogBackupFiles(), getMaxLogFileSize());
81  		RollingFileAppender appender = buildAppender(buildEncoder(), rollingPolicy);
82  		
83  		rollingPolicy.start();
84  		appender.start();
85  		
86  		// configure the logger for info and add the appender
87  		return getAndConfigureLogger(appender);
88  	}
89  	
90  	private RollingFileAppender buildAppender(Encoder encoder, RollingPolicy rollingPolicy) {
91  		RollingFileAppender appender = new RollingFileAppender();
92  		appender.setEncoder(encoder);
93  		appender.setAppend(true);
94  		appender.setContext(loggerContext);
95  		appender.setRollingPolicy(rollingPolicy);
96  		
97  		rollingPolicy.setParent(appender);
98  		return appender;
99  	}
100 	
101 	protected Logger getAndConfigureLogger(Appender appender) {
102 		ch.qos.logback.classic.Logger loggerLocal = loggerContext.getLogger(buildLoggerName());
103 		loggerLocal.setAdditive(false);
104 		loggerLocal.setLevel(Level.INFO);
105 		loggerLocal.addAppender(appender);
106 		return loggerLocal;
107 	}
108 	
109 	protected Encoder buildEncoder() {
110 		PatternLayoutEncoder logEncoder = new PatternLayoutEncoder();
111 		logEncoder.setContext(loggerContext);
112 		logEncoder.setPattern(getSettingOutputPattern());
113 		logEncoder.start();
114 		
115 		return logEncoder;
116 	}
117 				
118 	protected RollingPolicy initRollingPolicy(String fileName, int maxBackupFiles, String maxFileSize) {
119 		SizeAndTimeBasedFNATP sizeTimeBasedPolicy = new SizeAndTimeBasedFNATP();
120 		// the max file size before rolling to a new file
121 		sizeTimeBasedPolicy.setMaxFileSize(maxFileSize);
122 		sizeTimeBasedPolicy.setContext(loggerContext);
123 		
124 		TimeBasedRollingPolicy policy = new TimeBasedRollingPolicy();
125 		// set the filename pattern
126 		policy.setFileNamePattern(fileName);
127 		// the maximum number of backup files to keep around
128 		policy.setMaxHistory(maxBackupFiles);
129 		policy.setTimeBasedFileNamingAndTriggeringPolicy(sizeTimeBasedPolicy);
130 		policy.setContext(loggerContext);
131 		
132 		return policy;
133 	}
134 	
135 	protected String getSettingOutputPattern() {
136 		String outputPattern = (String) this.getSettings().get(SETTING_OUTPUT_PATTERN);
137 		if (outputPattern == null) {
138 			return DEFAULT_OUTPUT_PATTERN;
139 		}
140 		else {
141 			return outputPattern;
142 		}
143 	}
144 }
