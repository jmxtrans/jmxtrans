73  /**
74   * Represents a JMX Query to ask for obj, attr and one or more keys.
75   *
76   * @author jon
77   */
78  @JsonSerialize(include = NON_NULL)
79  @JsonPropertyOrder(value = {"obj", "attr", "typeNames", "resultAlias", "keys", "allowDottedKeys", "useAllTypeNames", "outputWriters"})
80  @ThreadSafe
81  @EqualsAndHashCode(exclude = {"outputWriters", "outputWriterInstances"})
82  @ToString(exclude = {"outputWriters", "typeNameValuesStringBuilder"})
83  public class Query {

93  	/**
94  	 * The list of type names used in a JMX bean string when querying with a
95  	 * wildcard which is used to expose the actual type name value to the key
96  	 * string. e.g. for this JMX name
97  	 * <p/>
98  	 * typeName=name=PS Eden Space,type=MemoryPool
99  	 * <p/>
100 	 * If you add a typeName("name"), then it'll retrieve 'PS Eden Space' from
101 	 * the string
102 	 */
103 	@Getter private final ImmutableSet<String> typeNames;

105 	/**
106 	 * The alias allows you to specify what you would like the results of the
107 	 * query to go into.
108 	 */
109 	@Getter private final String resultAlias;

111 	/**
112 	 * The useObjDomainAsKey property allows you to specify the use of the Domain portion of the Object Name
113 	 * as part of the output key instead of using the ClassName of the MBean which is the default behavior.
114 	 */
115 	@Getter private final boolean useObjDomainAsKey;
116 	@Getter private final boolean allowDottedKeys;
117 	@Getter private final boolean useAllTypeNames;
118 	@Nonnull @Getter private final ImmutableList<OutputWriterFactory> outputWriters;
119 	@Nonnull @Getter private final Iterable<OutputWriter> outputWriterInstances;
120 	private final TypeNameValuesStringBuilder typeNameValuesStringBuilder;

122 	@JsonCreator
123 	public Query(
124 			@JsonProperty("obj") String obj,
125 			@JsonProperty("keys") List<String> keys,
126 			@JsonProperty("attr") List<String> attr,
127 			@JsonProperty("typeNames") Set<String> typeNames,
128 			@JsonProperty("resultAlias") String resultAlias,
129 			@JsonProperty("useObjDomainAsKey") boolean useObjDomainAsKey,
130 			@JsonProperty("allowDottedKeys") boolean allowDottedKeys,
131 			@JsonProperty("useAllTypeNames") boolean useAllTypeNames,
132 			@JsonProperty("outputWriters") List<OutputWriterFactory> outputWriters
133 	) {
134 		try {
135 			this.objectName = new ObjectName(obj);
136 		} catch (MalformedObjectNameException e) {
137 			throw new IllegalArgumentException("Invalid object name: " + obj, e);
138 		}
139 		this.attr = copyOf(firstNonNull(attr, Collections.<String>emptyList()));
140 		this.resultAlias = resultAlias;
141 		this.useObjDomainAsKey = firstNonNull(useObjDomainAsKey, false);
142 		this.keys = copyOf(firstNonNull(keys, Collections.<String>emptyList()));
143 		this.allowDottedKeys = allowDottedKeys;
144 		this.useAllTypeNames = useAllTypeNames;
145 		this.outputWriters = outputWriters == null ? ImmutableList.<OutputWriterFactory>of() : copyOf(outputWriters);
146 		this.typeNames = ImmutableSet.copyOf(firstNonNull(typeNames, Collections.<String>emptySet()));
147 
148 		this.typeNameValuesStringBuilder = makeTypeNameValuesStringBuilder();
149 
150 		this.outputWriterInstances = createOutputWriters(outputWriters);
151 	}
152 
153 	private ImmutableList<OutputWriter> createOutputWriters(Iterable<OutputWriterFactory> outputWriters) {
154 		if (outputWriters == null) return ImmutableList.of();
155 		return FluentIterable
156 				.from(outputWriters)
157 				.transform(new Function<OutputWriterFactory, OutputWriter>() {
158 					@Nullable
159 					@Override
160 					public OutputWriter apply(OutputWriterFactory input) {
161 						return input.create();
162 					}
163 				})
164 				.toList();
165 	}

167 	public String makeTypeNameValueString(List<String> typeNames, String typeNameStr) {
168 		return this.typeNameValuesStringBuilder.build(typeNames, typeNameStr);
169 	}

175 	public Iterable<Result> fetchResults(MBeanServerConnection mbeanServer, ObjectName queryName) throws InstanceNotFoundException, IntrospectionException, ReflectionException, IOException {
176 		MBeanInfo info = mbeanServer.getMBeanInfo(queryName);
177 		ObjectInstance oi = mbeanServer.getObjectInstance(queryName);
178 
179 		List<String> attributes;
180 		if (attr.isEmpty()) {
181 			attributes = new ArrayList<String>();
182 			for (MBeanAttributeInfo attrInfo : info.getAttributes()) {
183 				attributes.add(attrInfo.getName());
184 			}
185 		} else {
186 			attributes = attr;
187 		}
188 
189 		try {
190 			if (!attributes.isEmpty()) {
191 				logger.debug("Executing queryName [{}] from query [{}]", queryName.getCanonicalName(), this);
192 
193 				AttributeList al = mbeanServer.getAttributes(queryName, attributes.toArray(new String[attributes.size()]));
194 
195 				return new JmxResultProcessor(this, oi, al.asList(), info.getClassName(), queryName.getDomain()).getResults();
196 			}
197 		} catch (UnmarshalException ue) {
198 			if ((ue.getCause() != null) && (ue.getCause() instanceof ClassNotFoundException)) {
199 				logger.debug("Bad unmarshall, continuing. This is probably ok and due to something like this: "
200 						+ "http://ehcache.org/xref/net/sf/ehcache/distribution/RMICacheManagerPeerListener.html#52", ue.getMessage());
201 			} else {
202 				throw ue;
203 			}
204 		}
205 		return ImmutableList.of();
206 	}

208 	private TypeNameValuesStringBuilder makeTypeNameValuesStringBuilder() {
209 		String separator = isAllowDottedKeys() ? "." : TypeNameValuesStringBuilder.DEFAULT_SEPARATOR;
210 		Set<String> typeNames = getTypeNames();
211 		if (isUseAllTypeNames()) {
212 			return new UseAllTypeNameValuesStringBuilder(separator);
213 		} else if (typeNames != null && !typeNames.isEmpty()) {
214 			return new PrependingTypeNameValuesStringBuilder(separator, new ArrayList<String>(typeNames));
215 		} else {
216 			return new TypeNameValuesStringBuilder(separator);
217 		}
218 	}

228 	public void runOutputWritersForQuery(Server server, Iterable<Result> results) throws Exception {
229 		for (OutputWriter writer : getOutputWriterInstances()) {
230 			writer.doWrite(server, this, results);
231 		}
232 		logger.debug("Finished running outputWriters for query: {}", this);
233 	}

235 	@NotThreadSafe
236 	@Accessors(chain = true)
237 	public static final class Builder {
238 		@Setter private String obj;
239 		private final List<String> attr = newArrayList();
240 		@Setter private String resultAlias;
241 		private final List<String> keys = newArrayList();
242 		@Setter private boolean useObjDomainAsKey;
243 		@Setter private boolean allowDottedKeys;
244 		@Setter private boolean useAllTypeNames;
245 		private final List<OutputWriterFactory> outputWriters = newArrayList();
246 		private final Set<String> typeNames = newHashSet();
247 
248 		private Builder() {}
249 
250 		/** This builder does NOT copy output writers from the given query. */
251 		private Builder(Query query) {
252 			this.obj = query.objectName.toString();
253 			this.attr.addAll(query.attr);
254 			this.resultAlias = query.resultAlias;
255 			this.keys.addAll(query.keys);
256 			this.useObjDomainAsKey = query.useObjDomainAsKey;
257 			this.allowDottedKeys = query.allowDottedKeys;
258 			this.useAllTypeNames = query.useAllTypeNames;
259 			this.typeNames.addAll(query.typeNames);
260 		}
261 
262 		public Builder addAttr(String... attr) {
263 			this.attr.addAll(asList(attr));
264 			return this;
265 		}
266 
267 		public Builder addKey(String keys) {
268 			return addKeys(keys);
269 		}
270 
271 		public Builder addKeys(String... keys) {
272 			this.keys.addAll(asList(keys));
273 			return this;
274 		}
275 
276 		public Builder addOutputWriter(OutputWriterFactory outputWriter) {
277 			return addOutputWriters(outputWriter);
278 		}
279 
280 		public Builder addOutputWriters(OutputWriterFactory... outputWriters) {
281 			this.outputWriters.addAll(asList(outputWriters));
282 			return this;
283 		}
284 
285 		public Builder addOutputWriters(Collection<OutputWriterFactory> outputWriters) {
286 			this.outputWriters.addAll(outputWriters);
287 			return this;
288 		}
289 
290 		public Builder setTypeNames(Set<String> typeNames) {
291 			this.typeNames.addAll(typeNames);
292 			return this;
293 		}
294 
295 		public Query build() {
296 			return new Query(
297 					this.obj,
298 					this.keys,
299 					this.attr,
300 					this.typeNames,
301 					this.resultAlias,
302 					this.useObjDomainAsKey,
303 					this.allowDottedKeys,
304 					this.useAllTypeNames,
305 					this.outputWriters
306 			);
307 		}
