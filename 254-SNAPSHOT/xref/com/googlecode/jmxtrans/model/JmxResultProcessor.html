41  import static com.google.common.collect.ImmutableList.Builder;
42  import static com.google.common.collect.Maps.newHashMap;

44  public class JmxResultProcessor {
45  
46  	private final Query query;
47  	private final ObjectInstance objectInstance;
48  	private final String className;
49  	private final String objDomain;
50  	private final List<Attribute> attributes;
51  
52  	public JmxResultProcessor(Query query, ObjectInstance objectInstance, List<Attribute> attributes, String className, String objDomain) {
53  		this.query = query;
54  		this.objectInstance = objectInstance;
55  		this.className = className;
56  		this.objDomain = objDomain;
57  		this.attributes = attributes;
58  	}
59  
60  	public ImmutableList<Result> getResults() {
61  		Builder<Result> accumulator = ImmutableList.builder();
62  		for (Attribute attribute : attributes) {
63  			getResult(accumulator, attribute);
64  		}
65  		return accumulator.build();
66  	}
67  
68  	/**
69  	 * Used when the object is effectively a java type
70  	 */
71  	private void getResult(Builder<Result> accumulator, Attribute attribute) {
72  		Object value = attribute.getValue();
73  		if (value == null) {
74  			return;
75  		}
76  
77  		if (value instanceof CompositeData) {
78  			getResult(accumulator, attribute.getName(), (CompositeData) value);
79  		} else if (value instanceof CompositeData[]) {
80  			for (CompositeData cd : (CompositeData[]) value) {
81  				getResult(accumulator, attribute.getName(), cd);
82  			}
83  		} else if (value instanceof ObjectName[]) {
84  			Map<String, Object> values = newHashMap();
85  			for (ObjectName obj : (ObjectName[]) value) {
86  				values.put(obj.getCanonicalName(), obj.getKeyPropertyListString());
87  			}
88  			Result r = getNewResultObject(attribute.getName(), values);
89  			accumulator.add(r);
90  		} else if (value.getClass().isArray()) {
91  			// OMFG: this is nutty. some of the items in the array can be
92  			// primitive! great interview question!
93  			Map<String, Object> values = newHashMap();
94  			for (int i = 0; i < Array.getLength(value); i++) {
95  				Object val = Array.get(value, i);
96  				values.put(attribute.getName() + "." + i, val);
97  			}
98  			accumulator.add(getNewResultObject(attribute.getName(), values));
99  		} else if (value instanceof TabularDataSupport) {
100 			TabularDataSupport tds = (TabularDataSupport) value;
101 			Map<String, Object> values = Collections.emptyMap();
102 			Result r = getNewResultObject(attribute.getName(), values);
103 			processTabularDataSupport(accumulator, attribute.getName(), tds);
104 			accumulator.add(r);
105 		}  else if (value instanceof Map) {
106 			Result r = getNewResultObject(attribute.getName(), convertKeysToString((Map<Object, Object>) value));
107 			accumulator.add(r);
108 		} else {
109 			Map<String, Object> values = newHashMap();
110 			values.put(attribute.getName(), value);
111 			Result r = getNewResultObject(attribute.getName(), values);
112 			accumulator.add(r);
113 		}
114 	}
115 
116 	private <K, V> ImmutableMap<String, V> convertKeysToString(Map<K, V> value) {
117 		ImmutableMap.Builder<String, V> values = ImmutableMap.builder();
118 		for (Map.Entry<K, V> entry : value.entrySet()) {
119 			values.put(entry.getKey().toString(), entry.getValue());
120 		}
121 		return values.build();
122 	}
123 
124 	/**
125 	 * Populates the Result objects. This is a recursive function. Query
126 	 * contains the keys that we want to get the values of.
127 	 */
128 	private void getResult(Builder<Result> accumulator, String attributeName, CompositeData cds) {
129 		CompositeType t = cds.getCompositeType();
130 
131 		Map<String, Object> values = newHashMap();
132 
133 		Set<String> keys = t.keySet();
134 		for (String key : keys) {
135 			Object value = cds.get(key);
136 			if (value instanceof TabularDataSupport) {
137 				TabularDataSupport tds = (TabularDataSupport) value;
138 				processTabularDataSupport(accumulator, attributeName + "." + key, tds);
139 				values.put(key, value);
140 			} else if (value instanceof CompositeDataSupport) {
141 				// now recursively go through everything.
142 				CompositeDataSupport cds2 = (CompositeDataSupport) value;
143 				getResult(accumulator, attributeName, cds2);
144 				return; // because we don't want to add to the list yet.
145 			} else {
146 				values.put(key, value);
147 			}
148 		}
149 		Result r = getNewResultObject(attributeName, values);
150 		accumulator.add(r);
151 	}
152 
153 	private void processTabularDataSupport(
154 			Builder<Result> accumulator, String attributeName,
155 			TabularDataSupport tds) {
156 		Set<Map.Entry<Object, Object>> entries = tds.entrySet();
157 		for (Map.Entry<Object, Object> entry : entries) {
158 			Object entryKeys = entry.getKey();
159 			if (entryKeys instanceof List) {
160 				// ie: attributeName=LastGcInfo.Par Survivor Space
161 				// i haven't seen this be smaller or larger than List<1>, but
162 				// might as well loop it.
163 				StringBuilder sb = new StringBuilder();
164 				for (Object entryKey : (List<?>) entryKeys) {
165 					sb.append(".");
166 					sb.append(entryKey);
167 				}
168 				String attributeName2 = sb.toString();
169 				Object entryValue = entry.getValue();
170 				if (entryValue instanceof CompositeDataSupport) {
171 					getResult(accumulator, attributeName + attributeName2, (CompositeDataSupport) entryValue);
172 				} else {
173 					throw new RuntimeException("!!!!!!!!!! Please file a bug: https://github.com/jmxtrans/jmxtrans/issues entryValue is: "
174 							+ entryValue.getClass().getCanonicalName());
175 				}
176 			} else {
177 				throw new RuntimeException("!!!!!!!!!! Please file a bug: https://github.com/jmxtrans/jmxtrans/issues entryKeys is: "
178 						+ entryKeys.getClass().getCanonicalName());
179 			}
180 		}
181 	}
182 
183 	/**
184 	 * Builds up the base Result object
185 	 */
186 	private Result getNewResultObject(String attributeName, Map<String, Object> values) {
187 		return new Result(System.currentTimeMillis(), attributeName, className, objDomain, query.getResultAlias(), objectInstance.getObjectName().getKeyPropertyListString(), values);
188 	}
