23  package com.googlecode.jmxtrans.model.naming;
24  
25  import com.googlecode.jmxtrans.model.NamingStrategy;
26  import com.googlecode.jmxtrans.model.Result;
27  import com.googlecode.jmxtrans.model.naming.typename.TypeNameValue;
28  import org.apache.commons.jexl2.Expression;
29  import org.apache.commons.jexl2.JexlContext;
30  import org.apache.commons.jexl2.JexlEngine;
31  import org.apache.commons.jexl2.JexlException;
32  import org.apache.commons.jexl2.MapContext;
33  import org.slf4j.Logger;
34  import org.slf4j.LoggerFactory;
35  
36  import java.util.Map;
37  
38  /**
39   * Naming strategy which uses an JEXL expression to format the name from the result, its class name, attribute name,
40   * and more.  Note the resulting name can be used for any of a number of purposes.  For example, with OpenTSDB it could
41   * be the metric name or a tag name.
42   * <p/>
43   * In the expression, all of the following may be used:
44   * <b>Variables</b>
45   * <dl>
46   * <dt>alias</dt>
47   * <dd>the classname alias of the MBean.</dd>
48   * <dt>attribute</dt>
49   * <dd>the name of the attribute of the MBean queried.</dd>
50   * <dt>class</dt>
51   * <dd>the effective classname of the MBean (uses #alias is not null, otherwise uses #realclass).</dd>
52   * <dt>realclass</dt>
53   * <dd>the class name of the MBean.</dd>
54   * <dt>result</dt>
55   * <dd>the full Result object.</dd>
56   * </dl>
57   */
58  public class JexlNamingStrategy implements NamingStrategy {
59  	private static final Logger LOG = LoggerFactory.getLogger(JexlNamingStrategy.class);
60  
61  	public static final String DEFAULT_EXPRESSION = "class + \".\" + attribute";
62  	public static final String VAR__CLASSNAME = "realclass";
63  	public static final String VAR__ATTRIBUTE_NAME = "attribute";
64  	public static final String VAR__CLASSNAME_ALIAS = "alias";
65  	public static final String VAR__EFFECTIVE_CLASSNAME = "class";
66  	public static final String VAR__TYPENAME = "typename";
67  	public static final String VAR__RESULT = "result";
68  
69  	protected JexlEngine jexl;
70  	protected Expression parsedExpr;
71  
72  
73  	/**
74  	 * Create a new naming strategy using an JEXL expression and the default expression.
75  	 */
76  	public JexlNamingStrategy() throws JexlException {
77  		jexl = new JexlEngine();
78  		this.parsedExpr = jexl.createExpression(DEFAULT_EXPRESSION);
79  	}
80  
81  	/**
82  	 * Create a new naming strategy using an JEXL expression and the given expression.
83  	 *
84  	 * @param expr - the JEXL expression to use to create names.
85  	 */
86  	public JexlNamingStrategy(String expr) throws JexlException {
87  		jexl = new JexlEngine();
88  		this.parsedExpr = jexl.createExpression(expr);
89  	}
90  
91  	/**
92  	 * Format the name for the given result.
93  	 *
94  	 * @param result - the result of a JMX query.
95  	 * @return String - the formatted string resulting from the expression, or null if the formatting fails.
96  	 */
97  	@Override
98  	public String formatName(Result result) {
99  		String formatted;
100 		JexlContext context = new MapContext();
101 
102 		this.populateContext(context, result);
103 		try {
104 			formatted = (String) this.parsedExpr.evaluate(context);
105 		} catch (JexlException jexlExc) {
106 			LOG.error("error applying JEXL expression to query results", jexlExc);
107 			formatted = null;
108 		}
109 
110 		return formatted;
111 	}
112 
113 	public void setExpression(String expr) throws JexlException {
114 		this.parsedExpr = this.jexl.createExpression(expr);
115 	}
116 
117 	/**
118 	 * Populate the context with values from the result.
119 	 *
120 	 * @param context - the expression context used when evaluating JEXL expressions.
121 	 * @param result  - the result of a JMX query.
122 	 */
123 	protected void populateContext(JexlContext context, Result result) {
124 		context.set(VAR__CLASSNAME, result.getClassName());
125 		context.set(VAR__ATTRIBUTE_NAME, result.getAttributeName());
126 		context.set(VAR__CLASSNAME_ALIAS, result.getKeyAlias());
127 
128 		Map<String, String> typeNameMap = TypeNameValue.extractMap(result.getTypeName());
129 		context.set(VAR__TYPENAME, typeNameMap);
130 
131 		String effectiveClassname = result.getKeyAlias();
132 		if (effectiveClassname == null) {
133 			effectiveClassname = result.getClassName();
134 		}
135 		context.set(VAR__EFFECTIVE_CLASSNAME, effectiveClassname);
136 
137 		context.set(VAR__RESULT, result);
138 	}
139 }
