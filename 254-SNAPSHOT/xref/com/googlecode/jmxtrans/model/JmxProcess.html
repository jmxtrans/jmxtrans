23  package com.googlecode.jmxtrans.model;
24  
25  import com.fasterxml.jackson.annotation.JsonIgnore;
26  import com.fasterxml.jackson.databind.annotation.JsonSerialize;
27  import com.google.common.collect.ImmutableList;
28  import lombok.Getter;
29  import lombok.Setter;
30  
31  import java.util.List;
32  
33  import static com.fasterxml.jackson.databind.annotation.JsonSerialize.Inclusion.NON_NULL;
34  
35  /**
36   * This is the container for a list of Servers.
37   *
38   * @author jon
39   */
40  @JsonSerialize(include = NON_NULL)
41  public class JmxProcess {
42  
43  	@Getter @Setter private String name;
44  	private ImmutableList<Server> servers;
45  	@Getter private Integer numMultiThreadedServers;
46  
47  	public JmxProcess() {
48  		this(ImmutableList.<Server>of());
49  	}
50  
51  	public JmxProcess(Server server) {
52  		this.servers = ImmutableList.of(server);
53  	}
54  
55  	public JmxProcess(ImmutableList<Server> servers) {
56  		this.setServers(servers);
57  	}
58  
59  	public void setServers(List<Server> servers) {
60  		this.servers = ImmutableList.copyOf(servers);
61  	}
62  
63  	public ImmutableList<Server> getServers() {
64  		return servers;
65  	}
66  
67  	@JsonIgnore
68  	public boolean isServersMultiThreaded() {
69  		return this.numMultiThreadedServers != null && this.numMultiThreadedServers > 0;
70  	}
71  
72  	/**
73  	 * Set this if you want each JmxProcess to run in its own thread up to the
74  	 * count you set. So, if you set this to 3 and you have 6 JmxProcesses,
75  	 * it'll start up a max of 3 threads and then block until a thread becomes
76  	 * available. It'll then start another thread up to fill in the blanks.
77  	 */
78  	public void setNumMultiThreadedServers(Integer numMultiThreadedServers) {
79  		this.numMultiThreadedServers = numMultiThreadedServers;
80  	}
81  
82  }
